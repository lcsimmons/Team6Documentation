import { Injectable } from '@angular/core';
import { Observable, tap } from 'rxjs';
import { environment } from '../environments/environment';
import { of } from 'rxjs';
import { HttpClient, HttpParams } from '@angular/common/http';

/**
 * Interface defining the structure of a menu item.
 */
export interface MenuItem {
  /** Numeric ID of the item */
  id: number;
  /** Name of the menu item */
  name: string; 
  /** Price of the item */
  price: number;
}

/**
 * Interface defining the structure for sell together data.
 */
export interface SellTogetherData {
  menu_item_1: number;
  menu_item_2: number;
  frequency: number;
}

/**
 * Interface defining the structure for employee data.
 */
interface Employee {
  id: number;
  name: string;
  title: string;
  salary: number;
  work_scheduling: string;
  permissions: number;
}

/**
 * Service that handles all data interactions for the application, such as fetching weather data,
 * managing cart items, and handling order histories.
 */
@Injectable({
  providedIn: 'root'
})
export class DataService {
  private readonly API_URL = 'https://api.openweathermap.org/data/3.0/onecall';

  /**
   * Construct data service.
   * @param http Httpclient to fulfil API requests.
   */
  constructor(private http: HttpClient) { }

   /**
   * Fetches the current weather data for a specific latitude and longitude.
   * @param lat The latitude coordinate.
   * @param lon The longitude coordinate.
   * @param apiKey The API key for the weather service.
   * @returns An Observable containing the weather data.
   */
  getWeather(lat: number, lon: number, apiKey: string): Observable<any> {
    const url = `${this.API_URL}?lat=${lat}&lon=${lon}&appid=${apiKey}`;
    return this.http.get(url);
  }

  private cart: MenuItem[] = [];

  /**
   * Adds a menu item to the shopping cart.
   * @param item The menu item to add.
   */
  addToCart(item: MenuItem): void {
    this.cart.push(item);
  }

  /**
   * Clears all items from the in-memory cart.
   */
  clearCart(): void {
    this.cart = [];
  }

   /**
   * Retrieves all menu items.
   * @returns An Observable of all menu items.
   */
  getMenuItems(): Observable<any> {
    return this.http.get(`${environment.apiUrl}/menu-items`);
  }

  /**
   * Retrieves menu items filtered by category.
   * @param category The category to filter by.
   * @returns An Observable of menu items in the specified category.
   */
  getMenuItemsByCategory(category: string): Observable<any> {
    return this.http.get(`${environment.apiUrl}/menu-categories`, { params: { category } });
  }

  /**
   * Retrieves the employee ID based on a given name.
   * @param name The name of the employee.
   * @returns An Observable containing the employee ID.
   */
  getEmployeeId(name: string): Observable<any> {
    return this.http.get(`${environment.apiUrl}/get-employee-id`, { params: { name } });
  }

  /**
   * Retrieves the current items in the shopping cart.
   * @returns An Observable array of current cart items.
   */
  getCartItems(): Observable<MenuItem[]> {
    return of(this.cart);
  }

  /**
   * Records an order history in the database.
   * @param order The order number.
   * @param itemList A list of item IDs.
   * @param togo Whether the order is to go.
   * @param totalPrice The total price of the order.
   * @param time The timestamp of the order.
   * @param employeeId The ID of the employee who took the order.
   * @param isCustomized Whether the order includes customized items.
   * @returns An Observable of the server response.
   */
  addOrderHistory(order: number, itemList: string, togo: boolean, totalPrice: number, time: string, employeeId: number, isCustomized: boolean): Observable<any> {
    const body = {
      order,
      itemList,
      togo,
      totalPrice,
      time,
      employeeId,
      isCustomized
    };
    return this.http.post(`${environment.apiUrl}/add_order_history`, body);
  }

  /**
   * Retrieves the next available order ID from the server.
   * @returns An Observable containing the next order ID.
   */
  getNextOrderId(): Observable<any> {
    return this.http.get(`${environment.apiUrl}/get_next_order_id`);
  }

  /**
   * Generate restock report data.
   * @param threshold limit of items before a item is considered in need of restock report
   * @returns Observable with restock report data.
   */
  getRestockReport(threshold: number): Observable<any> {
    return this.http.get(`${environment.apiUrl}/restock-report?threshold=${threshold}`);
  }

  getSupplier(): Observable<any> {
    return this.http.get(`${environment.apiUrl}/get-supplier`);
  }

  /**
   * Calculates the total cost of items in the cart, including a specified tax rate.
   * @returns The total cost including tax.
   */
  calculateTotalCost(): number {
    let totalCost = 0;
    for (let item of this.cart) {
      totalCost += item.price;
    }
    totalCost = parseFloat((totalCost * .0625 + totalCost).toFixed(2));
    return totalCost;
  }
  
  /**
   * Fetches the current seasonal item from the database.
   * @returns An Observable containing the seasonal item.
   */
  getSeasonalItem(): Observable<any> {
    return this.http.get(`${environment.apiUrl}/get_seasonal_item`);
  }

  /**
   * Removes an item from the cart based on its index.
   * @param index The index of the item in the cart.
   */
  removeItemFromCart(index: number): void {
    if (index > -1 && index < this.cart.length) {
      this.cart.splice(index, 1);
    }
  }

  /**
   * Fetches the ID of a menu item based on its name.
   * @param name The name of the menu item.
   * @returns An Observable containing the item ID.
   */
  getItemIdByItemName(name: string): Observable<any> {
    return this.http.get(`${environment.apiUrl}/get-item-id-by-name`, { params: { name } });
  }
  
  /**
   * Fetches a specific menu item by its ID.
   * @param id The ID of the menu item.
   * @returns An Observable containing the menu item.
   */
  getMenuItem(id: number): Observable<any> {
    return this.http.get(`${environment.apiUrl}/menu-item/${id}`);
  }


  /**
   * Submits ingredient changes for a specific menu item to the server.
   * @param payload The changes to submit.
   * @returns An Observable of the server response.
   */
  submitIngredientChanges(payload: any): Observable<any> {
    return this.http.post<any>(`${environment.apiUrl}/update-ingredients`, payload);
  }


 
  /**
   * Adds a new menu item to the server database.
   * @param menuItem The new menu item data.
   * @returns An Observable of the server response.
   */
  addMenuItem(menuItem: any): Observable<any> {
    return this.http.post<any>(`${environment.apiUrl}/add-menu-item`, menuItem);
  }
  
  /**
   * Retrieves all categories from the server.
   * @returns An Observable array of categories.
   */
  getCategories(): Observable<any> {
    return this.http.get<any>(`${environment.apiUrl}/categories`);
  }
  
  /**
   * Retrieves all ingredients from the server.
   * @returns An Observable array of ingredients.
   */
  getIngredients(): Observable<any> {
    return this.http.get<any>(`${environment.apiUrl}/ingredients`);
  } 

  /**
   * Fetches sales report data between two dates.
   * @param startDate The start date for the report.
   * @param endDate The end date for the report.
   * @returns An Observable containing the sales report data.
   */
  getSalesReport(startDate: string, endDate: string): Observable<any> {
    return this.http.get<any>(`${environment.apiUrl}/sales-report`, {
      params: { startDate, endDate }
    });
  }

  /**
   * Retrieves frequency data for items sold together within a specified date range.
   * @param startDateString The start date for the query.
   * @param endDateString The end date for the query.
   * @returns An Observable array of sell together frequency data.
   */
  getSellTogetherFrequency(startDateString: string, endDateString: string): Observable<any[]> {
    const url = `${environment.apiUrl}/sell-together-frequency`;
    const params = new HttpParams()
      .set('start_date', startDateString)
      .set('end_date', endDateString);
  
    return this.http.get<any[]>(url, { params });
  }
  

  /**
   * Fetches the difference in inventory between two dates.
   * @param startDate The start date for the inventory check.
   * @param endDate The end date for the inventory check.
   * @returns An Observable with the inventory difference data.
   */
  getInventoryDifference(startDate: string, endDate: string): Observable<any> {
    return this.http.get(`${environment.apiUrl}/get-inventory-difference`, {
      params: {
        start_date: startDate,
        end_date: endDate
      }
    });
  }

  /**
   * Retrieves the name of an ingredient based on its ID.
   * @param id The ID of the ingredient.
   * @returns An Observable containing the name of the ingredient.
   */
  getIngredientNameByID(id: number): Observable<any> {
    return this.http.get(`${environment.apiUrl}/get-ingredient-name-by-id`, { params: { id: id.toString() } });
  }

  /**
   * Retrieves the inventory for an ingredient by its name and record date.
   * @param ingredient_name The name of the ingredient.
   * @param record_date The date for the inventory record.
   * @returns An Observable containing the inventory data for the specified ingredient and date.
   */
  getIngredientInventoryByDate(ingredient_name: number, record_date: string): Observable<any> {
    return this.http.get(`${environment.apiUrl}/get-ingredient-inventory-by-date`, { 
      params: { 
        ingredient_name: ingredient_name.toString(), 
        record_date 
      } 
    });
  }

  // /==/=//=/=/==/=/

  addSeasonalMenuItem(item: any): Observable<any> {
    return this.http.post(`${environment.apiUrl}/add-seasonal-item`, item);
  }

  getlastSeasonalItem(): Observable<any> {
    return this.http.get(`${environment.apiUrl}/last-seasonal-item`);
  }

  deleteSeasonalItem(end_date: string): Observable<any> {
    return this.http.post(`${environment.apiUrl}/delete-seasonal-item`, { end_date });
  }

  // =/=/=//==/=//==//=/=/=


  getExcessReport(startDate: string): Observable<any> {
    return this.http.get<any>(`${environment.apiUrl}/excess-report`, {
      params: { startDate }
    });
  }

  

  saveCustomizations(orderId: number, menuItemId: number, ingredientIds: string): Observable<any> {
    const body = {
      orderId,
      menuItemId,
      ingredientIds
    };
    return this.http.post(`${environment.apiUrl}/save-customizations`, body);
  }

  getCustomizations(orderId: number, menuItemId: number): Observable<number[]> {
    const params = {
      orderId: orderId,
      menuItemId: menuItemId
    };
  
    return this.http.get<number[]>(`${environment.apiUrl}/get-customizations`, { params });
  }

  getMenuItemNameById(itemId: number): Observable<any> {
    return this.http.get<any>(`${environment.apiUrl}/get-menu-item-name-by-item-id`, { params: { item_id: itemId.toString() } });
  }


  getTopMenuItems(startDate: string, endDate: string, limitCount: number): Observable<any> {
    return this.http.get<any>(`${environment.apiUrl}/top-menu-items/${startDate}/${endDate}/${limitCount}`);
  }
