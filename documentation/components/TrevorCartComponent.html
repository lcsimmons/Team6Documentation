import { Component, OnInit, ViewChild } from '@angular/core';
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { DataService } from '../data.service';
import { concatMap } from 'rxjs/operators';
import { forkJoin } from 'rxjs';
/** Represent an item */
interface ItemId {
  /** Id number of the item in the database */
  itemId: string; // Adjust the type if itemId is not a string
}

/**
 * Acessible component to manage menu items in cart.
 */
@Component({
  selector: 'app-trevor-cart',
  templateUrl: './trevor-cart.component.html',
  styleUrl: './trevor-cart.component.css'
})
export class TrevorCartComponent implements OnInit{

  /**
   * Reference for the side navigation component.
   */
  @ViewChild('drawer') drawer!: MatSidenav;
  /**
   * Array of items in the cart.
   */
  cartItems: any[] = [];
  /**
   * Order number for the current order.
   */
  orderNumber: number = 0;
  /**
   *  Boolean to determine if the order is 'to-go'.
   */
  toGo: boolean = false;
  /**
   * List of item IDs in the cart.
   */
  itemListString: string = "";
  /**
   * Observable that emits true if the screen is a handset device, false otherwise.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );
  
  /**
   * Accessibility toolbar container visibility state.
   */
  isAccessibilityToolbarContainerVisible: boolean = true;

  /**
  * Toggles the visibility of the accessibility toolbar.
  */
  toggleAccessibilityToolbarContainer(): void {
    this.isAccessibilityToolbarContainerVisible = !this.isAccessibilityToolbarContainerVisible;
    console.log('Toolbar visibility:', this.isAccessibilityToolbarContainerVisible);
  }
  

  /**
   * Array of text sizes for the accessibility toolbar.
   */
  textsizers = [
    { size: 'Regular'},
    { size: 'Large'},
    { size: 'Extra Large'},
  ]

  /* end accessibility toolbar code */


  /**
   * Current weather description.
   */
  weatherDescription: string = '';
  /**
   * Current temperature.
   */
  temperature: number = 0;
  /**
   * Current 'feels like' temperature.
   */
  feelsLikeTemperature: number = 0;
  /**
   * Current weather icon.
   */
  weathericon: string = '01d.png';


  

  /**
  * Constructs the component with necessary dependencies.
  * @param dataService DataService to fetch and manage item data.
  * @param router Router for navigation.
  * @param breakpointObserver BreakpointObserver to respond to media query changes.
  */
  constructor(private dataService: DataService, private router: Router, private breakpointObserver: BreakpointObserver) { }

  /**
  * Initializes the component by fetching cart items and setting up external services.
  */
  ngOnInit(): void {
    this.dataService.getCartItems().subscribe(items => {
      this.cartItems = items;
    });
    this.callWeatherApi();
    this.loadGoogleTranslate();
  }

  /**
  * Calls a weather API to retrieve and display current weather conditions.
  */
  callWeatherApi(): void {
    const lat = 30.6125364
    const lon = -96.3433166
    const api_key = 'dcd3c687ea0ab81cb49468a99ae0ddb4'
    
    
    this.dataService.getWeather(lat, lon, api_key).subscribe((weather: any) => {
      this.weatherDescription = weather.current.weather[0].description;
      this.temperature = Math.round((weather.current.temp - 273.15) * (9 / 5) + 32);
      // this.feelsLikeTemperature = Math.round((weather.current.feels_like - 273.15) * (9 / 5) + 32);
      if(weather.current.weather[0].icon != null){
        if(weather.current.weather[0].icon == '01d'){
          this.weathericon = '01d.png';
        }
        if(weather.current.weather[0].icon == '01n'){
          this.weathericon = '01n.png';
        }
        if(weather.current.weather[0].icon == '02d'){
          this.weathericon = '02d.png';
        }
        if(weather.current.weather[0].icon == '02n'){
          this.weathericon = '02n.png';
        }
        if(weather.current.weather[0].icon == '03d' || weather.current.weather[0].icon == '03n'){
          this.weathericon = '03d.png';
        }
        if(weather.current.weather[0].icon == '04d' || weather.current.weather[0].icon == '04n'){
          this.weathericon = '04d.png';
        }
        if(weather.current.weather[0].icon == '09d' || weather.current.weather[0].icon == '09n'){
          this.weathericon = '09d.png';
        }
        if(weather.current.weather[0].icon == '10d'){
          this.weathericon = '10d.png';
        }
        if(weather.current.weather[0].icon == '10n'){
          this.weathericon = '10n.png';
        }
        if(weather.current.weather[0].icon == '11d' || weather.current.weather[0].icon == '11n'){
          this.weathericon = '11d.png';
        }
        if(weather.current.weather[0].icon == '13d' || weather.current.weather[0].icon == '13n'){
          this.weathericon = '13d.png';
        }
        if(weather.current.weather[0].icon == '50d' || weather.current.weather[0].icon == '50n'){
          this.weathericon = '50d.png';
        }
      }
      
    });

  }

  /**
  * Sets up Google Translate on the page.
  */
  loadGoogleTranslate(): void {
    const existingScript = document.getElementById('google-translate-script');
    if (!existingScript) {
      const script = document.createElement('script');
      script.id = 'google-translate-script';
      script.type = 'text/javascript';
      script.src = '//translate.google.com/translate_a/element.js?cb=googleTranslateElementInit';
      document.head.appendChild(script);
    }
      (window as any).googleTranslateElementInit = this.initGoogleTranslate;
  }

  /**
  * Initializes the Google Translate widget.
  */
  initGoogleTranslate(): void {
    new window.google.translate.TranslateElement({
      pageLanguage: 'en',
      layout: window.google.translate.TranslateElement.InlineLayout.SIMPLE
    }, 'google_translate_element');
  }

  /**
  * Toggles the 'to-go' status of the current order.
  */
  toggleToGo(): void {
    this.toGo = !this.toGo;
  }

  /**
  * Navigates to the item customization page.
  * @param itemId The ID of the item to customize.
  */
  editItem(itemId: number): void {
    console.log('Navigating to:', ['/trevor-customize-item', itemId]);
    this.router.navigate(['/trevor-customize-item', itemId]);
  }

  /**
  * Processes checkout, creates an order, and clears the cart.
  */
  checkout(): void {
    console.log("Checking out...");

    console.log("Items in cart:");
  
    this.dataService.getNextOrderId().pipe(
      concatMap((orderNumber: any) => {
        this.orderNumber = orderNumber.next_order_id;
  
        let observables = this.cartItems.map(item => {
          return this.dataService.getItemIdByItemName(item.name);
        });
        return forkJoin(observables);
      }),
      concatMap((itemIds: ItemId[]) => {
        this.itemListString = itemIds.map(item => item.itemId).join(", ");
  
        const totalPrice = this.dataService.calculateTotalCost();
        const time = new Date().toISOString();
        const employeeId = 1;
        const isCustomized = false;
  
        return this.dataService.addOrderHistory(this.orderNumber, this.itemListString, this.toGo, totalPrice, time, employeeId, isCustomized);
      })
    ).subscribe(() => {
      this.dataService.clearCart();
      this.cartItems = [];
      this.router.navigate(['/trevor-checkout']);
    });
  }

  /**
  * Removes an item from the cart.
  * @param index The index of the item in the cart array.
  */
  removeItem(index: number): void {
    if (index > -1 && index < this.cartItems.length) {
      const item = this.cartItems[index];
      this.cartItems.splice(index, 1);
      this.dataService.removeItemFromCart(item);
    }
  }

  
  /**
  * Calculates the total price of the items in the cart including taxes.
  * @returns The total price.
  */
  calculateTotalPrice(): number {
    const subtotal = this.cartItems.reduce((total, item) => total + item.price, 0);
    const taxes = this.calculateTaxes();
    return subtotal + taxes;
  }

  /**
  * Calculates the subtotal of the items in the cart.
  * @returns The subtotal amount.
  */
  calculateSubtotal(): number {
    return this.cartItems.reduce((total, item) => total + item.price, 0);
  }
  
  /**
  * Calculates the taxes based on the subtotal and a set tax rate.
  * @returns The calculated tax amount.
  */
  calculateTaxes(): number {
    const taxRate = 0.0625;
    return this.calculateSubtotal() * taxRate;
  }

  /**
  * Navigates back to the employee login screen.
  */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }
  
  /**
  * Navigates back to the menu screen.
  */
  onSeeMenu(): void {
    this.router.navigate(['/trevor-cashier']);
  }

}
