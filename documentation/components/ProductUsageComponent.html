  @ViewChild('drawer') drawer!: MatSidenav;
  date1: Date = new Date();
  date2: Date = new Date();
  date1String: string = this.date1.toISOString().split('T')[0];
  date2String: string = this.date2.toISOString().split('T')[0];
  stockData: any;
  displayedColumns: string[] = ['productUsed', 'startInventory', 'endInventory', 'productDifference'];

  categories: string[] = ['Burger', 'Sandwich', 'Value Meal', 'Basket', 'Side', 'Shakes & More', 'Beverage', 'Extras', 'Sauces'];

  dataSource = new MatTableDataSource<any>();

  /**
  * Observable to track if the device is a handset for responsive layout purposes.
  */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  constructor(private dataService: DataService, private router: Router, private breakpointObserver: BreakpointObserver) { }

  /**
   * Initializes component data and settings on component load.
   */
  ngOnInit(): void {
    this.stockData = [
      { start: '', end: '', startInventory: 0, endInventory: 0 }
    ];
  }

  /**
   * Updates the date range for inventory checks based on user selection and triggers table update.
   * @param event HTML select element event containing the selected range.
   */
  updateDateRange(event: Event) {
    const target = event.target as HTMLSelectElement;
    const range = target.value;
  
    let startDate: Date = new Date();
    let endDate = new Date();
  
    switch (range) {
      case 'month':
        startDate = new Date(endDate.getFullYear(), endDate.getMonth() - 1, endDate.getDate());
        break;
      case 'year':
        startDate = new Date(endDate.getFullYear() - 1, endDate.getMonth(), endDate.getDate());
        break;
      case '2weeks':
        startDate = new Date(endDate.getFullYear(), endDate.getMonth(), endDate.getDate() - 14);
        break;
      default:
        startDate = this.date1;
        endDate = this.date2;
    }

    this.date1 = startDate;
    this.date2 = endDate;

    this.date1String = this.date1.toISOString().split('T')[0];
    this.date2String = this.date2.toISOString().split('T')[0];
  
    this.updateTable(this.date1String, this.date2String);
  }

  /**
   * Fetches and updates the table with inventory differences based on the selected date range.
   * @param startDate The starting date of the period for the inventory check.
   * @param endDate The ending date of the period for the inventory check.
   */
  updateTable(startDate: string, endDate: string): void {
    const startDateISO = new Date(startDate).toISOString().split('T')[0];
    const endDateISO = new Date(endDate).toISOString().split('T')[0];

    console.log('Start Date: ', startDateISO);
    console.log('End Date: ', endDateISO);

    this.dataService.getInventoryDifference(startDateISO, endDateISO).subscribe(data => {
      forkJoin(
        Object.keys(data).map(key => 
          forkJoin({
            ingredientName: this.dataService.getIngredientNameByID(Number(key)).pipe(map(res => res.ingredientName)),
            startInventory: this.dataService.getIngredientInventoryByDate(Number(key), startDateISO).pipe(map(res => res.quantity || 'Not found')),
            endInventory: this.dataService.getIngredientInventoryByDate(Number(key), endDateISO).pipe(map(res => res.quantity || 'Not found'))

          }).pipe(
            map(response => {
              const productDifference = response.startInventory - response.endInventory;
              return {
                productUsed: response.ingredientName,
                productDifference: productDifference,
                startInventory: response.startInventory,
                endInventory: response.endInventory
              };
            })
          )
        )
      ).subscribe(stockData => {
        // Filter out items with productDifference < 0
        const filteredStockData = stockData.filter(item => item.productDifference >= 0);

        this.stockData = filteredStockData;
        this.dataSource = new MatTableDataSource(this.stockData);
        this.dataSource.sort = this.sort;
        this.chartData = filteredStockData.map(item => ({
          name: item.productUsed,
          series: [
            { name: 'Start Inventory', value: item.startInventory },
            { name: 'End Inventory', value: item.endInventory }
          ]
        }));
      });
    });
  }


  /**
   * Navigates back to the employee login screen.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }

  /**
   * Toggles the Bionic theme on or off.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
  }
}
