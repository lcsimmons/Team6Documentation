import { Component } from '@angular/core';
import { Router } from '@angular/router';

/**
 * Component for managing the main login page and navigation based on user roles.
 */
@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.css']
})
export class LoginComponent {
  /**
   * Constructs the login component with dependency injection for the router.
   * @param router Router service to manage navigation.
   */
  constructor(private router: Router) {}

  /**
   * Navigates to the menu board, typically for management or staff use.
   */
  navigateToMenuBoard(): void {
    this.router.navigate(['/menu-board']);
  }

  /**
   * Navigates to the customer ordering system, intended for customer interactions.
   */
  navigateToCustomerOrdering(): void {
    this.router.navigate(['/customer-landing']);
  }

  /**
   * Navigates to the employee login page, for employee access.
   */
  navigateToEmployeeLogin(): void {
    this.router.navigate(['/employee-login']);
  }
}
