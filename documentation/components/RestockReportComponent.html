import { Component } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { DataService } from '../data.service';
import { DatePipe } from '@angular/common';
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { MatSnackBar } from '@angular/material/snack-bar';

/**
 * Interface defining the structure for inventory items.
 */
interface InventoryItem {
  id: number;
  name: string;
  price: number;
  quantity: number;
  currentQuantity: number; // Add currentQuantity property
  quantityToAdd: number;   // Add quantityToAdd property
  sliderValue: number; // Add sliderValue property
}

/**
 * Component for managing and reporting inventory restocking.
 */
@Component({
  selector: 'app-restock-report',
  templateUrl: './restock-report.component.html',
  styleUrls: ['./restock-report.component.css']
})
export class RestockReportComponent {

  Bionic = false;

  /**
  * Observable to track if the device is a handset for responsive layout purposes.
  */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  ingredientName: string = '';
  currentQuantity: number = 0;
  quantityToAdd: number = 0;
  inventory: InventoryItem[] = [];

  quantityThreshold: number = 100;

  /**
   * Constructs the Restock Report component with necessary dependencies.
   * @param dataService DataService for fetching and updating inventory data.
   * @param router Router for navigation.
   * @param breakpointObserver BreakpointObserver for responsive adjustments.
   * @param _snackBar MatSnackBar for displaying notifications.
   */
  constructor(private dataService: DataService, private router: Router, private breakpointObserver: BreakpointObserver, private _snackBar: MatSnackBar) { }

  restockItems: any[] = [];

  /**
   * Initializes the component by fetching initial restock data and inventory.
   */
  ngOnInit() {
    this.fetchRestockReport();
    this.fetchInventory();
  }

  /**
   * Navigates back to the employee login screen.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }

  /**
   * Toggles the current visual theme of the component.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
  }

  /**
   * Updates the current quantity displayed on the UI.
   */
  updateCurrentQuantity(): void {
    console.log("updateCurrentQuantity called");
    const selectedIngredient = this.inventory.find(item => item.name === this.ingredientName);
    if (selectedIngredient) {
      this.currentQuantity = selectedIngredient.quantity;
    }
  }

  /**
   * Initializes slider values based on current inventory quantities.
   */
  initializeSliderValues(): void {
    this.inventory.forEach(item => {
      item.sliderValue = item.quantity;
    });
  }
  
  /**
   * Fetches restock report data from the DataService.
   */
  fetchRestockReport(): void {
    console.log('fetchRestockReport called with threshold:', this.quantityThreshold);
    
    this.dataService.getRestockReport(this.quantityThreshold).subscribe(
      (data: any[]) => {
        console.log('Received restock report data:', data);
        this.restockItems = data;
        console.log('Updated restockItems:', this.restockItems);
      },
      error => {
        console.error('Error fetching restock report:', error);
        this.openSnackBar('Failed to fetch restock report. Please try again later.');
      }
    );
  }
  
  /**
   * Fetches inventory data from the DataService.
   */
  fetchInventory(): void {
    console.log("fetchInventory called");
    this.dataService.getIngredientInventory().subscribe(
      (inventory: InventoryItem[]) => {
        console.log("Inventory fetched successfully:", inventory);
        this.inventory = inventory;
        if (this.inventory.length > 0) {
          this.ingredientName = this.inventory[0].name;
          this.currentQuantity = this.inventory[0].quantity;
        }        
      },
      error => {
        console.error('Error fetching inventory:', error);
        this.openSnackBar('Failed to fetch inventory. Please try again later.');
      }
    );
  }

  /**
   * Updates the inventory for a specific ingredient.
   */
  updateInventory(): void {
    console.log("updateInventory called");
    const selectedIngredient = this.inventory.find(item => item.name === this.ingredientName);
    if (selectedIngredient) {
      const updatedQuantity = selectedIngredient.quantity + this.quantityToAdd;
  
      if (this.quantityToAdd <= 0) {
        this.openSnackBar("Quantity must be positive to restock.");
      }
  
      this.dataService.updateIngredientQuantity(selectedIngredient.name, updatedQuantity).subscribe(
        (response: any) => {
          console.log('Quantity updated successfully:', response);
          selectedIngredient.quantity = updatedQuantity;
          this.quantityToAdd = 0;
          this.updateCurrentQuantity(); 
          this.fetchRestockReport(); 
        },
        (error: any) => {
          console.error('Error updating quantity:', error);
          this.openSnackBar('Failed to update quantity. Please try again later.');
        }
      );
    } else {
      console.error('Selected ingredient not found in inventory');
      this.openSnackBar('Selected ingredient not found in inventory');
    }
  }

  /**
   * Opens a snack bar with a specified message.
   * @param message The message to display.
   */
  openSnackBar(message: string) {
    this._snackBar.open(message, "Done", {
      duration: 5000
    });
  }

}
