import { Component, OnInit, ViewChild, ViewEncapsulation } from '@angular/core';//, OnDestroy
import { DataService } from '../data.service';
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { MatSnackBar } from '@angular/material/snack-bar';
// import { ViewEncapsulation } from '@angular/compiler';


declare global {
  /** Window to support Google maps */
  interface Window {
    /** Allows interaction with Google API */
    google: any;
  }
}
/**
 * Interface to define menu items
 */
export interface MenuItem {
  /** Numeric ID of the item */
  id: number;
  /** Name of the menu item */
  name: string; 
  /** Price of the item */
  price: number;
  /** How many are being ordered */
  quantity: number;
}

/**
 * Component for displaying and interacting with the customer menu,
 * including managing categories, items, and accessibility features.
 */
@Component({
  selector: 'app-customer-menu',
  templateUrl: './customer-menu.component.html',
  styleUrl: './customer-menu.component.css'
})

export class CustomerMenuComponent implements OnInit{//, OnDestroy {

  /* accesibility toolbar code here */
  isAccessibilityToolbarContainerVisible: boolean = true;

  /**
   * Toggles the visibility of the accessibility toolbar to adjust text sizes.
   */
  toggleAccessibilityToolbarContainer(): void {
    this.isAccessibilityToolbarContainerVisible = !this.isAccessibilityToolbarContainerVisible;
    console.log('Toolbar visibility:', this.isAccessibilityToolbarContainerVisible);
  }
  

  /**
   * Text size options for accessibility features.
   */
  textsizers = [
    { size: 'Normal', cssClass: 'text-normal' },
    { size: 'Large', cssClass: 'text-large' },
    { size: 'Extra Large', cssClass: 'text-extra-large' }
  ];

  /* end accessibility toolbar code */

  /**
   * weather description variable
   */
  weatherDescription: string = '';
  /**
   * temperature variable
   */
  temperature: number = 0;
  // feelsLikeTemperature: number = 0;
  /**
   * weather icon variable
   */
  weathericon: string = '01d.png';

  /**
   * Reference to the side navigation drawer.
   */
  @ViewChild('drawer') drawer!: MatSidenav;
  /**
   * Array of menu items to display in the customer view.
   */
  menuItems: MenuItem[] = [];
  /**
   * The currently selected menu item for customization or addition to cart.
   */
  selectedMenuItem?: MenuItem; 
  /**
   * The currently selected category of menu items.
   */
  selectedCategory?: string;
  /**
   * Array of categories for menu items.
   */
  categories: string[] = ['Burger', 'Sandwich', 'Value Meal', 'Basket', 'Side', 'Shakes & More', 'Beverage', 'Extras', 'Sauces', 'Seasonal'];

  /**
   * Initializes the component with required services.
   * @param dataService 
   * @param router 
   * @param breakpointObserver 
   * @param _snackBar 
   */
  constructor(private dataService: DataService, private router: Router, private breakpointObserver: BreakpointObserver, private _snackBar: MatSnackBar) { }

  /**
   * Observable to determine if the device is a handset for responsive layout purposes.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

    /**
     * Sets up Google Translate on the page.
     */
    loadGoogleTranslate(): void {
      const existingScript = document.getElementById('google-translate-script');
      if (!existingScript) {
        const script = document.createElement('script');
        script.id = 'google-translate-script';
        script.type = 'text/javascript';
        script.src = '//translate.google.com/translate_a/element.js?cb=googleTranslateElementInit';
        document.head.appendChild(script);
      }
        (window as any).googleTranslateElementInit = this.initGoogleTranslate;
    }
  
    /**
     * Initializes the Google Translate widget.
     */
    initGoogleTranslate(): void {
      new window.google.translate.TranslateElement({
        pageLanguage: 'en',
        layout: window.google.translate.TranslateElement.InlineLayout.SIMPLE
      }, 'google_translate_element');
    }
  
    /**
     * Initializes the component by setting a default category and loading external services.
     */
    ngOnInit(): void {
      this.onSelectCategory("Burger");
      this.callWeatherApi();
      this.loadGoogleTranslate();
    }

    /**
     * Calls a weather API to retrieve and display current weather conditions.
     */
    callWeatherApi(): void {
      const lat = 30.6125364
      const lon = -96.3433166
      const api_key = 'dcd3c687ea0ab81cb49468a99ae0ddb4'
      
      
      this.dataService.getWeather(lat, lon, api_key).subscribe((weather: any) => {

        console.log(weather.current.weather[0].icon);

        this.weatherDescription = weather.current.weather[0].description;
        this.temperature = Math.round((weather.current.temp - 273.15) * (9 / 5) + 32);
        // this.feelsLikeTemperature = Math.round((weather.current.feels_like - 273.15) * (9 / 5) + 32);
        if(weather.current.weather[0].icon != null){
          if(weather.current.weather[0].icon == '01d'){
            this.weathericon = '01d.png';
          }
          if(weather.current.weather[0].icon == '01n'){
            this.weathericon = '01n.png';
          }
          if(weather.current.weather[0].icon == '02d'){
            this.weathericon = '02d.png';
          }
          if(weather.current.weather[0].icon == '02n'){
            this.weathericon = '02n.png';
          }
          if(weather.current.weather[0].icon == '03d' || weather.current.weather[0].icon == '03n'){
            this.weathericon = '03d.png';
          }
          if(weather.current.weather[0].icon == '04d' || weather.current.weather[0].icon == '04n'){
            this.weathericon = '04d.png';
          }
          if(weather.current.weather[0].icon == '09d' || weather.current.weather[0].icon == '09n'){
            this.weathericon = '09d.png';
          }
          if(weather.current.weather[0].icon == '10d'){
            this.weathericon = '10d.png';
          }
          if(weather.current.weather[0].icon == '10n'){
            this.weathericon = '10n.png';
          }
          if(weather.current.weather[0].icon == '11d' || weather.current.weather[0].icon == '11n'){
            this.weathericon = '11d.png';
          }
          if(weather.current.weather[0].icon == '13d' || weather.current.weather[0].icon == '13n'){
            this.weathericon = '13d.png';
          }
          if(weather.current.weather[0].icon == '50d' || weather.current.weather[0].icon == '50n'){
            this.weathericon = '50d.png';
          }
        }
        
      });
  
    }

  /**
   * Fetches menu items for a selected category.
   * @param category The category to fetch items for.
   */
  onSelectCategory(category: string): void {
    this.selectedCategory = category;
    this.dataService.getMenuItemsByCategory(category).subscribe((items: any[]) => {
      // Ensure that each item has an 'id' property
      this.menuItems = items.map((item: any) => {
        return { id: item.id, name: item.name, price: Number(item.price), quantity: 1 };
      });
    });
  }

  /**
   * Selects a menu item for further actions like adding to cart.
   * @param item The menu item to select.
   */
  onSelectMenuItem(item: MenuItem): void {
    this.selectedMenuItem = item;
    // console.log('Selected item ID:', item.id)
  }


  /**
   * Determines if a menu item is currently selected.
   * @param item 
   * @returns 
   */
  isItemSelected(item: MenuItem): boolean {
    return this.selectedMenuItem?.name === item.name;
  }

  /**
   * Determines if a category is currently selected.
   * @param category 
   * @returns 
   */
  isCategorySelected(category: string): boolean {
    return this.selectedCategory == category;
  }

  /**
   * Navigates to the customer cart page.
   */
  onSeeCart(): void {
    this.router.navigate(['/customer-cart']);
  }

  /**
   * Adds the selected menu item to the shopping cart multiple times based on its quantity.
   */
  addToCart(): void {
    if (this.selectedMenuItem && this.selectedMenuItem.quantity > 0) {
      for (let i = 0; i < this.selectedMenuItem.quantity; i++) {
        this.dataService.addToCart(this.selectedMenuItem);
      }
      this.selectedMenuItem = undefined;
      this.openSnackBar("Item added to cart.");
    }
  }

  /**
   * Creates a combo by adding a default set of items to the cart along with the selected item.
   */
  makeCombo(): void {
    if (this.selectedMenuItem) {
      // Fetch fries and 20 oz fountain drink items from data
      const friesItem: MenuItem = { id: 14, name: 'French Fries', price: 1.99, quantity: 1 };
      const drinkItem: MenuItem = { id: 25, name: '20 oz. Fountain Drink', price: 1.99, quantity: 1 };

      // Add selected item, fries, and drink to the cart
      this.dataService.addToCart(this.selectedMenuItem);
      this.dataService.addToCart(friesItem);
      this.dataService.addToCart(drinkItem);
      this.openSnackBar("Item updated as a combo.");
    }
  }
  
  /**
   * Logs out the user and clears the cart.
   */
  goHome(): void {
    this.dataService.clearCart();
    this.router.navigate(['/employee-login']);
  }

  /**
   * Displays a message using a snack bar.
   * @param message The message to display.
   */
  openSnackBar(message: string) {
    this._snackBar.open(message, "Done", {
      duration: 5000
    });
  }

  // Toggle text size 
  

  /**
   * The currently selected text size for the page.
   */
  selectedTextSize: string = 'text-normal'; // Default size

  /**
   * Sets the text size for the page.
   * @param cssClass 
   */
  setTextSize(cssClass: string): void {
    this.selectedTextSize = cssClass;
  }

  

}
