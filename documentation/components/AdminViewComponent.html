import { Component, OnInit } from '@angular/core';
import { AuthService } from '../auth.service';
import { DataService } from '../data.service';
import { MatDialog } from '@angular/material/dialog';
import { EditEmployeeDialogComponent } from '../edit-employee-dialog/edit-employee-dialog.component';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { AddEmployeeComponent } from '../add-employee/add-employee.component';

/**
 * Interface for employee objects.
 */
export interface Employee {
  /** Numeric ID */
  id: number;
  /** Employee name */
  name: string;
  /** Employee title */
  title: string;
  /** Employee Salary */
  salary: number;
  /** Work schedule */
  work_scheduling: string;
  /** Level of permission on application */
  permissions: number;
}

/**
 * Component for managing the administrative view of employees.
 */
@Component({
  selector: 'app-admin-view',
  templateUrl: './admin-view.component.html',
  styleUrls: ['./admin-view.component.css']
})

/**
 * Component responsible for the admin view functionality.
 */
export class AdminViewComponent implements OnInit {

  /** Flag to track current style */
  Bionic = false;

  /**
   * Observable to track if the device is a handset for responsive layout purposes.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );
  
  /**
   * List of employees to be displayed.
   */
  employees: Employee[] = [];

  /**
   * Columns to display in the employee management table.
   */
  displayedColumns: string[] = ['email', 'title', 'salary', 'work_scheduling', 'permissions', 'update', 'delete'];

  /**
 * Constructs the AdminViewComponent with necessary dependencies.
 * @param authService The authentication service.
 * @param dataService The data service for fetching data.
 * @param dialog The dialog service for displaying dialogs.
 * @param breakpointObserver The observer for breakpoints.
 */
  constructor(private authService: AuthService, private dataService: DataService, private dialog: MatDialog, private breakpointObserver: BreakpointObserver) { }

  /**
   * Initializes the component by fetching employee data.
   */
  ngOnInit(): void {
    this.fetchEmployees();
  }

  /**
   * Toggles the visual theme of the application.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
  }

  /**
   * Logs out the current user.
   */
  logout(): void {
    this.authService.logout();
  }

  /**
   * Opens a dialog to edit an employee's details.
   * @param employee The employee to edit.
   */
  openEditDialog(employee: Employee): void {
    const dialogRef = this.dialog.open(EditEmployeeDialogComponent, {
      data: { ...employee }
    });
  
    dialogRef.afterClosed().subscribe(() => {
      this.fetchEmployees();
    });
  }

  /**
   * Fetches all employees from the server.
   */  
  fetchEmployees(): void {
    this.dataService.getEmployees().subscribe(
      employees => {
        this.employees = employees;
      },
      error => {
        console.error('Error fetching employees:', error);
      }
    );
  }

  /**
   * Opens a dialog to add a new employee.
   */
  addEmployee(): void {
    const dialogRef = this.dialog.open(AddEmployeeComponent, {
      width: '500px',
      data: {}
    });
  
    dialogRef.afterClosed().subscribe(() => {
      this.fetchEmployees();
    });
  }

  /**
   * Deletes an employee from the server.
   * @param employee The employee to delete.
   */
  deleteEmployee(employee: Employee) {
    this.dataService.deleteEmployee(employee.id).subscribe({
        next: response => {
            console.log(response);
            this.fetchEmployees();
        },
        error: error => {
            console.error(error);
        }
    });
}
}
