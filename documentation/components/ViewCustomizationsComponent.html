import { Component, Inject, OnInit } from '@angular/core';
import { MAT_DIALOG_DATA } from '@angular/material/dialog';
import { DataService } from '../data.service';
import { concatMap, map, toArray } from 'rxjs/operators';
import { from } from 'rxjs';

/**
 * Component to display customizations for a specific order within a dialog.
 */
@Component({
  selector: 'app-view-customizations',
  templateUrl: './view-customizations.component.html',
  styleUrls: ['./view-customizations.component.css']
})
export class ViewCustomizationsComponent implements OnInit {
  /**
   * The order data passed to the dialog.
   */
  order: any;
  /**
   * Array to hold the customizations details for the menu items in the order.
   */
  customizations: any[] = [];

  /**
   * Initializes the component with data injection and a reference to the DataService.
   * @param data The data passed to the dialog, typically the order details.
   * @param dataService The service used to fetch data from the server.
   */
  constructor(
    @Inject(MAT_DIALOG_DATA) public data: any,
    private dataService: DataService
  ) {
    this.order = data;
  }

  /**
   * On initialization, fetches customizations for each menu item in the order.
   */
  ngOnInit() {
    // Get the customizations for the order
    //console.log('Order:', this.order);  
  
    const menuItems$ = from(this.order.menu_items);
  
    menuItems$.pipe(
      concatMap((menuItem: any) => {
        return this.dataService.getItemIdByItemName(menuItem.menu_item_name).pipe(
          map(response => {
            const itemId = response.itemId - 100;
            return itemId;
          }),
          concatMap((itemId: number) => this.dataService.getCustomizations(this.order.id, itemId)),
          concatMap((customizations: any[]) => {
            // Convert each customization id to its name
            const customizationNames$ = from(customizations).pipe(
              concatMap((id: number) => {
                const adjustedId = id + 200; 
                //console.log('Adjusted ID:', adjustedId);
                return this.dataService.getIngredientNameByID(adjustedId);
              }),
              map(response => {
                //console.log('Response:', response); 
                return response.ingredientName;
              }),
              toArray()
            );
            return customizationNames$.pipe(
              map((customizationNames: string[]) => {
                return { menuItemName: menuItem.menu_item_name, customizations: customizationNames };
              })
            );
          })
        );
      })
    ).subscribe(({ menuItemName, customizations }: { menuItemName: string, customizations: string[] }) => {
      this.customizations.push({ menuItemName, customizations });
      //console.log('Customizations:', this.customizations);
    });
  }
}
