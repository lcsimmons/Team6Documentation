import { Component, Inject } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { DataService } from '../data.service';
import { concatMap, from, toArray } from 'rxjs';

/**
 * Component to handle the removal of items from an existing order.
 */
@Component({
  selector: 'app-remove-order-items-dialog',
  templateUrl: './remove-order-items-dialog.component.html',
  styleUrl: './remove-order-items-dialog.component.css'
})
export class RemoveOrderItemsDialogComponent {
  /**
   * Form to submit edits to the order. 
   */
  orderForm: FormGroup;

  /**
   * Array to hold items in the current order. 
   */
  menuItems: any[] = [];

  /**
   * Constructs the RemoveOrderItemsDialogComponent with injected dependencies and initial data setup.
   * @param data The data passed to the dialog, containing the order details.
   * @param fb FormBuilder to create form controls for the dialog.
   * @param dialogRef Reference to the dialog that opened this component.
   * @param dataService Service for data operations related to orders.
   */
  constructor(
    @Inject(MAT_DIALOG_DATA) public data: any,
    private fb: FormBuilder,
    public dialogRef: MatDialogRef<RemoveOrderItemsDialogComponent>,
    private dataService: DataService
  ) {
    // Parse the menu_items string into an array
    console.log(data.menu_items);
    this.menuItems = JSON.parse(data.menu_items);
  
    // Create a form group with a checkbox for each item in the order
    const formControls = this.menuItems.reduce((controls: any, item: any, index: number) => {
      controls[index] = this.fb.control(false);
      return controls;
    }, {});
  
    this.orderForm = this.fb.group(formControls);
  }

  /**
   * Submits the form to remove selected items from the order and update the order details.
   */
  onFormSubmit() {
    if (this.orderForm.valid) {
      const selectedItems = Object.keys(this.orderForm.value)
        .filter(index => this.orderForm.value[index])
        .map(index => this.menuItems[Number(index)]); // Explicitly cast index to number
  
      //console.log('Selected menu items for deletion:', selectedItems);
  
      // Call the removeOrderItems method from the DataService
      this.dataService.removeOrderItems(this.data.id, selectedItems).subscribe(response => {
        //console.log(response);
  
        // Calculate the total price of the remaining items
        let remainingItems = [...this.menuItems]; // Create a copy of this.menuItems
        let selectedItemsCopy = [...selectedItems]; // Create a copy of selectedItems
  
        for (let item of selectedItemsCopy) {
          const index = remainingItems.indexOf(item);
          if (index !== -1) {
            remainingItems.splice(index, 1); // Remove the item from remainingItems
          }
        }
  
        console.log('Remaining items:', remainingItems);
        let totalPrice = 0;
        from(remainingItems).pipe(
          concatMap((itemId: unknown) => this.dataService.getMenuItem((itemId as number) - 100)), // Decrement itemId by 100
          toArray()
        ).subscribe(menuItems => {
          totalPrice = menuItems.reduce((total, item) => total + item.price, 0);
  
          // Round the total price to two decimal places
          totalPrice = Number((totalPrice + totalPrice * .0625).toFixed(2));
          //console.log('Total price of remaining items:', totalPrice);
  
          // Prepare the data to be sent to updateOrder
          const updateData = {
            menuItems: [], // Send the remaining items after deletion
            togo: this.data.togo, // Assuming this.data.togo contains the previous togo value
            employeeId: this.data.employee_id, // Assuming this.data.employeeId contains the previous employee ID
            totalPrice: totalPrice
          };
  
          //console.log('Data being sent to updateOrder:', updateData);
  
          // Update the order with the new total price
          this.dataService.updateOrder(this.data.id, updateData)
            .subscribe(updateResponse => {
              //console.log(updateResponse);
              this.dialogRef.close(selectedItems);
            });
        }, error => {
          console.error(error);
        });
      });
    }
  }
}
