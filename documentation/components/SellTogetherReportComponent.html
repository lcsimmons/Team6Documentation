import { Component } from '@angular/core';
import { DataService } from '../data.service';
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { DatePipe } from '@angular/common'
import { FormBuilder, FormGroup } from '@angular/forms';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable, forkJoin, timer, of } from 'rxjs';
import { map, shareReplay, mergeMap, catchError } from 'rxjs/operators';
import { MatSnackBar } from '@angular/material/snack-bar';
import { MatSort } from '@angular/material/sort';
import { ViewChild } from '@angular/core';
import { MatTableDataSource } from '@angular/material/table';

/**
 * Interface defining the structure for sell together data.
 */
export interface SellTogetherData {
  /** Item one */
  menu_item_1: string;
  /** Item 2 */
  menu_item_2: string;
  /** Frequency */
  frequency: number;
}

/**
 * Component for generating reports on frequently sold items together.
 */
@Component({
  selector: 'app-sell-together-report',
  templateUrl: './sell-together-report.component.html',
  styleUrls: ['./sell-together-report.component.css'],
  providers: [DatePipe]
})
export class SellTogetherReportComponent {

  /**
   * Bionic variable to toggle the visual theme of the component.
   */
  Bionic = false;

  // startDate: Date;
  // endDate: Date;
  // range: string = '';
  /**
   * Form group for the date range selection form.
   */
  dateForm!: FormGroup;
  /**
   * Array of sell together data to display in the report.
   */
  sellTogetherData: SellTogetherData[] = [];
  /**
   * Columns to display in the sell together report table.
   */
  displayedColumns: string[] = ['menu_item_1', 'menu_item_2', 'frequency'];

  /**
   * Observable to determine if the device is a handset for responsive layout purposes.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

    /**
     * Reference to the sort component for the table.
     */
  @ViewChild(MatSort, {static: false}) sort!: MatSort;
  /**
   * Data source for the sell together report table.
   */
  dataSource = new MatTableDataSource<any>(this.sellTogetherData);
  

  /**
   * Constructs the Sell Together Report component with necessary dependencies.
   * @param fb FormBuilder to handle form creation and validation.
   * @param datePipe DatePipe for formatting dates.
   * @param dataService DataService to fetch and submit data.
   * @param router Router for navigation.
   * @param breakpointObserver BreakpointObserver to respond to media query changes.
   * @param _snackBar MatSnackBar to display notifications.
   */
  constructor(private fb: FormBuilder, private datePipe: DatePipe, private dataService: DataService, private router: Router, private breakpointObserver: BreakpointObserver, private _snackBar: MatSnackBar) {

    // Initialize startDate and endDate with current date
    // this.startDate = new Date();
    // this.endDate = new Date();
    this.dateForm = this.fb.group({
      startDate: [''],
      endDate: [''],
      range: ['']
    });
  }

  /**
   * Opens a snack bar with a custom message.
   * @param message The message to display.
   */
  openSnackBar(message: string) {
    this._snackBar.open(message, "Done", {
      duration: 5000
    });
  }

  /**
   * Fetches data for items that are frequently sold together within a specified date range.
   */
  getSellTogetherData(): void {
    // Update the date logic as needed...
    let { startDate, endDate, range } = this.dateForm.value;
  
    let actualStartDate: Date;
    let actualEndDate: Date;
    const now = new Date();
  
    // Determine the actual start and end dates based on the selected range
    switch (range) {
      case 'month':
        actualStartDate = new Date(now.getFullYear(), now.getMonth() - 1, now.getDate());
        actualEndDate = now;
        break;
      case 'year':
        actualStartDate = new Date(now.getFullYear() - 1, now.getMonth(), now.getDate());
        actualEndDate = now;
        break;
      case '2weeks':
        actualStartDate = new Date(now.getFullYear(), now.getMonth(), now.getDate() - 14);
        actualEndDate = now;
        break;
      default:
        actualStartDate = startDate ? new Date(startDate) : new Date();
        actualEndDate = endDate ? new Date(endDate) : new Date();
    }

    // Check if the start date is valid
    if (actualStartDate > actualEndDate) {
      this.openSnackBar("Error: Start date cannot be before end date");
      return; 
    }

    // Formatting the dates to ISO strings for the API
    const startDateString = actualStartDate.toISOString();
    const endDateString = actualEndDate.toISOString();

    this.dataService.getSellTogetherFrequency(startDateString, endDateString)
      .pipe(
        mergeMap((data: SellTogetherData[]) => {
          const batchRequests: Observable<SellTogetherData>[] = [];
          data.forEach(item => {
            const menu1$ = this.fetchMenuItemNameById(Number(item.menu_item_1));
            const menu2$ = this.fetchMenuItemNameById(Number(item.menu_item_2));
            batchRequests.push(forkJoin([menu1$, menu2$]).pipe(
              map(([menu1, menu2]) => ({
                ...item,
                menu_item_1: menu1,
                menu_item_2: menu2
              })),
              catchError(() => of(null as unknown as SellTogetherData)) // Handle individual request errors gracefully
            ));
          });
          return forkJoin(batchRequests);
        })
      )
      .subscribe(
        (updatedData: (SellTogetherData | null)[]) => {
          this.sellTogetherData = updatedData.filter(data => !!data) as SellTogetherData[]; // Filter out null values
          this.dataSource = new MatTableDataSource(this.sellTogetherData);
          this.dataSource.sort = this.sort;
        },
        (error: any) => { // Define the type of error as any or Error
          this.openSnackBar("Error fetching sell together data");
        }
      );
  }

  /**
   * Fetches the name of a menu item by its ID.
   * @param itemId 
   * @returns 
   */
  fetchMenuItemNameById(itemId: number): Observable<string> { // Change parameter type to number
    // Add delay here if necessary
    return this.dataService.getMenuItemNameById(itemId).pipe(
      map(menu => menu?.menu_item_name || ''), // Extract menu_item_name or return empty string
      catchError(() => of('')) // Handle request error gracefully
    );
  }

  /**
   * Formats a date to an ISO string format.
   * @param date The date to format.
   * @returns The formatted date string.
   */
  private formatDate(date: Date): string {
    return date.toISOString().split('T')[0];
  }

  /**
   * Navigates back to the employee login screen.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }

  /**
   * Toggles the current visual theme of the component.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
  }
}
