import { Component, OnInit,  ChangeDetectorRef } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { DataService } from '../data.service';

import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { Router } from '@angular/router';

/**
 * Interface to define the structure of an ingredient object.
 */
export interface Ingredient {
  id: number;
  name: string;
  price: number;
}


/**
 * Component for managing the editing of ingredients within the restaurant's management system.
 */
@Component({
  
  selector: 'app-manager-edit-ingredients',
  templateUrl: './manager-edit-ingredients.component.html',
  styleUrls: ['./manager-edit-ingredients.component.css']
})
export class ManagerEditIngredientsComponent implements OnInit {
  
  /** Flag for toggling style dynamically within the component. */
  Bionic = false;

  /** Holds the name of the current supplier. */
  supplierName: string = '';

  /** List of ingredients to display and manage. */
  ingredients: Ingredient[] = [];

  /** Currently selected ingredient's ID for editing. */
  selectedIngredientId: number | null = null;

  /** Form for editing an existing ingredient. */
  ingredientForm: FormGroup;

  /** Form for creating a new ingredient. */
  newIngredientForm: FormGroup;

  /** Observable to determine if the device is a handset for responsive layout purposes. */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  constructor(private cdr: ChangeDetectorRef, private fb: FormBuilder, private dataService: DataService, private breakpointObserver: BreakpointObserver, private router: Router) {
    
    this.ingredientForm = this.fb.group({
      name: ['', Validators.required],
      price: ['', [Validators.required, Validators.min(0)]]
    });
    this.newIngredientForm = this.fb.group({
      name: ['', Validators.required],
      price: ['', [Validators.required, Validators.min(0)]]
    });
  }



  ngAfterViewInit() {
    this.cdr.detectChanges();
  }

  /**
   * Initializes the component by loading ingredients and supplier data.
   */
  ngOnInit(): void {
    this.loadIngredients();
    this.dataService.getSupplier().subscribe(data => {
      if (data && data.length > 0) {
        this.supplierName = data[0].supplier; // Assumes that data always contains at least one supplier
      }
    }, error => {
      console.error('Failed to load supplier:', error);
    });
  }

  /**
   * Loads all ingredients from the backend and sets the first as the selected ingredient.
   */
  loadIngredients(): void {
    this.dataService.getIngredients().subscribe(data => {
        console.log("Ingredients loaded:", data);  // Log the loaded data
        this.ingredients = data;
        if (data.length > 0) {
            this.selectedIngredientId = data[0].id;
            this.loadSelectedIngredient();
        }
    });
}


  /**
   * Loads the selected ingredient's details into the ingredient form if the form has not been modified.
   */
  loadSelectedIngredient(): void {
    if (!this.ingredientForm.dirty) {  // Only load if no changes have been made to the form
      let ingredient = this.ingredients.find(i => i.id === this.selectedIngredientId);
      if (ingredient) {
        this.ingredientForm.patchValue({
          name: ingredient.name,
          price: ingredient.price
        });
      }
    }
  }


  // loadSelectedIngredient(): void {   

  //   let ingredient = this.ingredients.find(i => i.id == this.selectedIngredientId);
  //   console.log("Found Ingredient:", ingredient);
    
  //   if (ingredient) {
  //     this.ingredientForm.patchValue({
  //       name: ingredient.name,
  //       price: ingredient.price
  //     });
  //   } else {
  //     console.error('No ingredient found with the ID:', this.selectedIngredientId);
  //   }
  // }
  

  /**
   * Submits the form to update an ingredient and reloads ingredients on success.
   */
  submitChanges(): void {
    if (this.selectedIngredientId) {
      this.dataService.updateIngredient(this.selectedIngredientId, this.ingredientForm.value.name, this.ingredientForm.value.price).subscribe({
        
        next: (res) => {
          console.log('item id', this.selectedIngredientId);
          console.log('new name', this.ingredientForm.value.name);
          console.log('new price', this.ingredientForm.value.price);
          console.log('Ingredient updated successfully', res);
          this.loadIngredients();
        },
        error: (err) => console.error('Error updating ingredient', err)
      });
    }
  }

  /**
   * Adds a new ingredient using the values from the newIngredientForm.
   */
  addNewIngredient(): void {
    if (this.newIngredientForm.valid) {
      this.dataService.addIngredient(this.newIngredientForm.value.name, this.newIngredientForm.value.price).subscribe({
        next: (res) => {
          console.log('new name', this.newIngredientForm.value.name);
          console.log('new price', this.newIngredientForm.value.price);
          console.log('Ingredient added successfully', res);
          this.loadIngredients();
        },
        error: (err) => console.error('Error adding ingredient', err)
      });
    }
  }

  /**
   * Removes the selected ingredient from the database.
   */
  removeIngredient(): void {
    if (this.selectedIngredientId !== null) {
      this.dataService.removeIngredientFromTable(this.selectedIngredientId).subscribe(
        response => {
          console.log('Deletion successful', response);
          this.loadIngredients();
        },
        error => {
          console.error('Deletion failed', error);
        }
      );
  } else {
    console.error('No ingredient item selected for deletion');
  }
  }

  /**
   * Navigates back to the employee login page.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }


  /**
   * Toggles the Bionic style theme and forces the component to check for changes.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
    this.cdr.markForCheck();  
    this.cdr.detectChanges();
  }


}
