import { Component, OnInit, Inject } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { DataService } from '../data.service';
import { MatDialogRef, MAT_DIALOG_DATA } from '@angular/material/dialog';


/**
 * Component for editing an employee's details through a dialog.
 */
@Component({
  selector: 'app-edit-employee-dialog',
  templateUrl: './edit-employee-dialog.component.html',
  styleUrls: ['./edit-employee-dialog.component.css']
})
export class EditEmployeeDialogComponent implements OnInit {
  /**
   * Form group for editing an employee.
   */
  editEmployeeForm: FormGroup;

  /**
   * Employee ID, used for updating the correct record.
   */
  employee_id: number;

  /**
   * Constructs the component with necessary dependencies and initializes the form.
   * @param formBuilder FormBuilder to create the form group.
   * @param dataService DataService to handle data operations.
   * @param dialogRef Reference to the dialog that launched this component.
   * @param data Data injected into the dialog, typically contains the employee being edited.
   */
  constructor(
    private formBuilder: FormBuilder, 
    private dataService: DataService, 
    public dialogRef: MatDialogRef<EditEmployeeDialogComponent>,
    @Inject(MAT_DIALOG_DATA) public data: any
  ) {
    this.employee_id = data.id;
    this.editEmployeeForm = this.formBuilder.group({
      name: [data.name, Validators.required],
      title: [{value: data.title, disabled: true}, Validators.required],
      salary: [data.salary, Validators.required],
      work_scheduling: [data.work_scheduling, Validators.required],
      permissions: [data.permissions, Validators.required]
    });
  }

  /**
   * Initializes the component.
   */
  ngOnInit(): void {
  }

  /**
   * Handles form submission to update an employee. Validates the form, constructs the employee data, and uses the data service to update the employee.
   */
  onSubmit(): void {
    if (this.editEmployeeForm.valid) {
      let employeeData = { id: this.employee_id, ...this.editEmployeeForm.value };
  
      // Convert permissions to a number
      employeeData.permissions = Number(employeeData.permissions);
  
      // Print the value of employeeData.permissions
      console.log('Permissions before switch: ', employeeData.permissions);
  
      // Update the title based on the permissions
      switch (employeeData.permissions) {
        case 1:
          employeeData.title = 'cashier';
          break;
        case 2:
          employeeData.title = 'manager';
          break;
        case 3:
          employeeData.title = 'admin';
          break;
        default:
          console.log('Invalid permissions');
          return;
      }
  
      // Print the value of employeeData after the switch
      console.log('Employee data after switch: ', employeeData);
  
      // Update the employee
      this.dataService.updateEmployee(employeeData).subscribe(response => {
        console.log(response);
        this.dialogRef.close();
      });
    }
  }
}
