import { Component, OnInit, ViewChild, ElementRef } from '@angular/core';
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { DataService } from '../data.service';
import { concatMap } from 'rxjs/operators';
import { forkJoin } from 'rxjs';

/**
 * Used to interact with the Google Maps API
 */
declare var google: any;

@Component({
  selector: 'app-customer-landing',
  templateUrl: './customer-landing.component.html',
  styleUrls: ['./customer-landing.component.css']
})
export class CustomerLandingComponent implements OnInit{
  
      /* accesibility toolbar code here */
      isAccessibilityToolbarContainerVisible: boolean = true;

      toggleAccessibilityToolbarContainer(): void {
        this.isAccessibilityToolbarContainerVisible = !this.isAccessibilityToolbarContainerVisible;
        console.log('Toolbar visibility:', this.isAccessibilityToolbarContainerVisible);
      }
      
       /**
       * Array of text size options available for screen display.
       */
      textsizers = [
        { size: 'Normal', cssClass: 'text-normal' },
        { size: 'Large', cssClass: 'text-large' },
        { size: 'Extra Large', cssClass: 'text-extra-large' }
        ];
    
        /**
         * The currently selected text size for display.
         */
        selectedTextSize: string = 'text-normal';
        
        /**
         * Sets the text size based on the selected CSS class.
         * @param cssClass The CSS class that defines the text size.
         */
        setTextSize(cssClass: string): void {
          this.selectedTextSize = cssClass;
        }
      /* end accessibility toolbar code */
    
    
      weatherDescription: string = '';
      temperature: number = 0;
      feelsLikeTemperature: number = 0;
      weathericon: string = '01d.png';
    
  constructor(private router: Router){}
      
  ngOnInit(): void {}
  
  display: any;
  center: google.maps.LatLngLiteral = {
      lat: 30.6125502,
      lng: -96.3413613
  };
  zoom = 15;

  // markerOptions: google.maps.MarkerOptions = { draggable: false };
  markerPosition: google.maps.LatLngLiteral = {
    lat: 30.6125502,  // Specific latitude for the marker
    lng: -96.3413613  // Specific longitude for the marker
  };
  
  /**
   * Updates the center of the map to the location where the user clicked.
   * @param event The map click event containing the new location.
   */
  moveMap(event: google.maps.MapMouseEvent) {
      if (event.latLng != null) this.center = (event.latLng.toJSON());
  }

  
  move(event: google.maps.MapMouseEvent) {
      if (event.latLng != null) this.display = event.latLng.toJSON();
  }  

  GoToOrder(): void {
    console.log("Going to customer menu...");
    this.router.navigate(['/customer-menu']);
  }

  scrollTo(id: string) {
    const element = document.getElementById(id);
    if (element) {
        element.scrollIntoView({ behavior: 'smooth', block: 'start' });
    }
}
}
