import { Component } from '@angular/core';
import { FormBuilder, FormGroup } from '@angular/forms';
import { DataService } from '../data.service';
import { DatePipe } from '@angular/common'
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import {MatSnackBar} from '@angular/material/snack-bar';
import { MatSort } from '@angular/material/sort';
import { ViewChild } from '@angular/core';
import { MatTableDataSource } from '@angular/material/table';

/**
 * Component for displaying sales reports.
 */
@Component({
  selector: 'app-sales-report',
  templateUrl: './sales-report.component.html',
  styleUrls: ['./sales-report.component.css'],
  providers: [DatePipe]
})
export class SalesReportComponent {
  /**
   * Array of sales data to display in the report.
   */
  salesData: any[] = [];
  /**
   * Sorting component for the sales data table.
   */
  @ViewChild(MatSort, {static: false}) sort!: MatSort;
  /**
   * Data source for the sales data table.
   */
  dataSource = new MatTableDataSource<any>(this.salesData);

  /**
   * Bionic variable to toggle the visual theme of the component.
   */
  Bionic = false;

  /**
   * Form group for the date range selection form.
   */
  dateForm!: FormGroup;


  /**
   * Creates a new chartData object to store the sales data for the chart.
   */
  public chartData: { name: string, series: { name: string, value: number }[] }[] = [];

  /**
  * Observable to track if the device is a handset for responsive layout purposes.
  */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  /**
   * Constructs the Sales Report component with necessary dependencies.
   * @param fb FormBuilder for building the date form.
   * @param router Router for navigation.
   * @param dataService DataService for fetching sales data.
   * @param datePipe DatePipe for formatting dates.
   * @param breakpointObserver BreakpointObserver for responsive adjustments.
   * @param _snackBar MatSnackBar for displaying notifications.
   */
  constructor(private fb: FormBuilder, private router: Router, private dataService: DataService, private datePipe: DatePipe, private breakpointObserver: BreakpointObserver, private _snackBar: MatSnackBar) {
      this.dateForm = this.fb.group({
      startDate: [''],
      endDate: [''],
      range: ['']
    });
  }

  /**
   * Initializes the component by preparing resources.
   */
  ngOnInit() {
    
  }


  /**
   * Displays a snack bar message.
   * @param message The message to be displayed.
   */
  openSnackBar(message: string) {
    this._snackBar.open(message, "Done", {
      duration: 5000
    });
  }

  /**
   * Handles form submission to generate a sales report based on selected dates.
   */
  onSubmit(): void {
    let { startDate, endDate, range } = this.dateForm.value;
  
    let actualStartDate: Date;
    let actualEndDate: Date;
    const now = new Date();
  
    switch (range) {
      case 'month':
        actualStartDate = new Date(now.getFullYear(), now.getMonth() - 1, now.getDate());
        actualEndDate = now;
        break;
      case 'year':
        actualStartDate = new Date(now.getFullYear() - 1, now.getMonth(), now.getDate());
        actualEndDate = now;
        break;
      case '2weeks':
        actualStartDate = new Date(now.getFullYear(), now.getMonth(), now.getDate() - 14);
        actualEndDate = now;
        break;
      default:
        actualStartDate = startDate;
        actualEndDate = now;
    }
  
    if (actualStartDate > actualEndDate) {
      this.openSnackBar("Error: Start date cannot be before end date");
      return; 
    }
  
    if (actualStartDate && actualEndDate) {
      const formattedStartDate = this.datePipe.transform(actualStartDate, 'MM-dd-yyyy');
      const formattedEndDate = this.datePipe.transform(actualEndDate, 'MM-dd-yyyy');
      if (formattedStartDate && formattedEndDate) {
        this.dataService.getSalesReport(formattedStartDate, formattedEndDate).subscribe({
          next: (data) => {
            this.salesData = data;
            this.chartData = this.salesData.map(item => ({
              name: item.menu_item_name,
              series: [
                { name: 'Items Sold', value: item.number_sold }
              ]
            }));
            this.dataSource = new MatTableDataSource(this.salesData);
            this.dataSource.sort = this.sort;
          },
          error: (err) => console.error('Failed to get sales report:', err)
        });        
      }
    } else {
      this.openSnackBar("Error: Start date and end date required");
      return;
    }
  }

  /**
   * Navigates back to the employee login screen.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }

  /**
   * Toggles the current visual theme of the component.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
  }
  
}
