import { Component, OnInit } from '@angular/core';
import { DataService } from '../data.service';
import { Router } from '@angular/router';

/**
 * Component for displaying the menu board with different categories.
 */
@Component({
  selector: 'app-menu-board',
  templateUrl: './menu-board.component.html',
  styleUrls: ['./menu-board.component.css']
})
export class MenuBoardComponent implements OnInit {
  /**
   * Array to hold categories with their respective items.
   */
  displayCategories: any[] = [];

  /**
   * Predefined categories available for selection.
   */
  categories: string[] = ['Burger', 'Sandwich', 'Value Meal', 'Basket', 'Side', 'Shakes & More', 'Beverage', 'Extras', 'Sauce', 'Seasonal'];

  /**
   * Initializes a new instance of the MenuBoardComponent.
   * @param dataService Service to fetch menu items by category.
   * @param router Router service to navigate between components.
   */
  constructor(private dataService: DataService, private router: Router) { }

  /**
   * Lifecycle hook that is called after Angular has initialized all data-bound properties.
   */
  ngOnInit(): void {
    this.categories.forEach(category => {
      this.onSelectCategory(category);
    });
  }

  /**
   * Retrieves items by category from the displayCategories array.
   * @param category The category to filter items by.
   * @returns An array of items belonging to the specified category.
   */
  getItemsByCategory(category: string): any[] {
    const categoryObject = this.displayCategories.find(cat => cat.category === category);
    return categoryObject ? categoryObject.items : [];
  }
  

  /**
   * Selects a category and fetches its menu items from the DataService, then adds them to the displayCategories array.
   * @param category The category to fetch menu items for.
   */
  onSelectCategory(category: string): void {
    this.dataService.getMenuItemsByCategory(category).subscribe(items => {
      let categoryItems = [];
      for (let name in items) {
        categoryItems.push({ name: name, price: items[name] });
      }
      this.displayCategories.push({ category: category, items: categoryItems }); 
    });
  }

  /**
   * Navigates back to the employee login page.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }
}
