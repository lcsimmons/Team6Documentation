import { Component } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { DataService } from '../data.service'; 
import { FormControl } from '@angular/forms';
import { Router } from '@angular/router';

import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { concatMap } from 'rxjs/operators';
import { forkJoin } from 'rxjs';

/**
 * Component for customizing a specific menu item selected by the customer.
 */
@Component({
  selector: 'app-trevor-customize-item',
  templateUrl: './trevor-customize-item.component.html',
  styleUrl: './trevor-customize-item.component.css'
})

/**
 * Component for customizing menu items in the Trevor cashier interface.
 */
export class TrevorCustomizeItemComponent {

  /** Details of the menu item being customized. */
  menuItem: any;

  /** List of ingredients available for customization. */
  ingredients: any[] = [];

  /** Selected ingredients for customization. */
  selectedIngredients: { [key: number]: FormControl } = {};

  /** Current order number. */
  orderNumber: number = 0;

  /** ID of the order being customized. */
  orderId: number | null = null; 

  /**
   * Observable to determine if the device is a handset for responsive layout purposes.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );
  
  /**
   * Toggles the visibility of the accessibility toolbar container.
   */
  isAccessibilityToolbarContainerVisible: boolean = true;

  /**
   * Toggles the visibility of the accessibility toolbar container.
   */
  toggleAccessibilityToolbarContainer(): void {
    this.isAccessibilityToolbarContainerVisible = !this.isAccessibilityToolbarContainerVisible;
    console.log('Toolbar visibility:', this.isAccessibilityToolbarContainerVisible);
  }
  
  /** Text sizes available for customization. */
  textsizers = [
    { size: 'Regular'},
    { size: 'Large'},
    { size: 'Extra Large'},
  ]

  /* end accessibility toolbar code */


  /** Weather description. */
  weatherDescription: string = '';

  /** Current temperature. */
  temperature: number = 0;

  /** Feels like temperature. */
  feelsLikeTemperature: number = 0;

  /** Weather icon. */
  weathericon: string = '01d.png';

  /**
   * Constructs the component with necessary dependencies.
   * @param route ActivatedRoute to access route parameters.
   * @param dataService DataService to fetch and submit data related to the menu items.
   * @param router Router for navigation.
   * @param breakpointObserver BreakpointObserver to react to media query changes.
   */
  constructor(
    private route: ActivatedRoute, 
    private dataService: DataService,
    private router: Router,
    private breakpointObserver: BreakpointObserver
  ) {}

  /**
   * Initalize the component
   */
  ngOnInit(): void {
    this.dataService.getNextOrderId().subscribe((response: any) => {
      this.orderNumber = response.next_order_id; 

    });
  
    const itemId = this.route.snapshot.paramMap.get('id');
    if (itemId !== null) {
      this.dataService.getMenuItem(+itemId).subscribe((menuItem: any) => {
        this.menuItem = menuItem;
        this.ingredients = menuItem.ingredients;
        this.selectedIngredients = {}; 
        this.ingredients.forEach(ingredient => {
          this.selectedIngredients[ingredient.id] = new FormControl(true);
        });
  
        this.loadCustomizations();
      });
    } else {
      console.error('ID parameter is null');
    }

    this.callWeatherApi();  
    this.loadGoogleTranslate();
  }

  /**
   * Calls the API to get current weather conditions for a fixed location.
   */
  callWeatherApi(): void {
    const lat = 30.6125364
    const lon = -96.3433166
    const api_key = 'dcd3c687ea0ab81cb49468a99ae0ddb4'
    
    
    this.dataService.getWeather(lat, lon, api_key).subscribe((weather: any) => {
      this.weatherDescription = weather.current.weather[0].description;
      this.temperature = Math.round((weather.current.temp - 273.15) * (9 / 5) + 32);
      // this.feelsLikeTemperature = Math.round((weather.current.feels_like - 273.15) * (9 / 5) + 32);
      if(weather.current.weather[0].icon != null){
        if(weather.current.weather[0].icon == '01d'){
          this.weathericon = '01d.png';
        }
        if(weather.current.weather[0].icon == '01n'){
          this.weathericon = '01n.png';
        }
        if(weather.current.weather[0].icon == '02d'){
          this.weathericon = '02d.png';
        }
        if(weather.current.weather[0].icon == '02n'){
          this.weathericon = '02n.png';
        }
        if(weather.current.weather[0].icon == '03d' || weather.current.weather[0].icon == '03n'){
          this.weathericon = '03d.png';
        }
        if(weather.current.weather[0].icon == '04d' || weather.current.weather[0].icon == '04n'){
          this.weathericon = '04d.png';
        }
        if(weather.current.weather[0].icon == '09d' || weather.current.weather[0].icon == '09n'){
          this.weathericon = '09d.png';
        }
        if(weather.current.weather[0].icon == '10d'){
          this.weathericon = '10d.png';
        }
        if(weather.current.weather[0].icon == '10n'){
          this.weathericon = '10n.png';
        }
        if(weather.current.weather[0].icon == '11d' || weather.current.weather[0].icon == '11n'){
          this.weathericon = '11d.png';
        }
        if(weather.current.weather[0].icon == '13d' || weather.current.weather[0].icon == '13n'){
          this.weathericon = '13d.png';
        }
        if(weather.current.weather[0].icon == '50d' || weather.current.weather[0].icon == '50n'){
          this.weathericon = '50d.png';
        }
      }
      
    });

  }

  /**
   * Opens Google Translate feature for the application.
   */
  loadGoogleTranslate(): void {
    const existingScript = document.getElementById('google-translate-script');
    if (!existingScript) {
      const script = document.createElement('script');
      script.id = 'google-translate-script';
      script.type = 'text/javascript';
      script.src = '//translate.google.com/translate_a/element.js?cb=googleTranslateElementInit';
      document.head.appendChild(script);
    }
      (window as any).googleTranslateElementInit = this.initGoogleTranslate;
  }

  /**
   * Initializes the Google Translate widget.
   */
  initGoogleTranslate(): void {
    new window.google.translate.TranslateElement({
      pageLanguage: 'en',
      layout: window.google.translate.TranslateElement.InlineLayout.SIMPLE
    }, 'google_translate_element');
  }
  
  /**
   * Loads customizations for the current order and menu item.
   */
  loadCustomizations(): void {
    console.log('Loading customizations for order ID:', this.orderNumber);
    console.log('Menu item ID:', this.menuItem.id);
  
    if (this.orderNumber !== 0 && this.menuItem !== undefined) {
      this.dataService.getCustomizations(this.orderNumber, this.menuItem.id).subscribe((customizations: any[]) => {
        console.log('Customizations loaded:', customizations);
        customizations.forEach(customization => {
          const ingredientId = customization;
          if (this.selectedIngredients[ingredientId]) {
            this.selectedIngredients[ingredientId].setValue(true);
          }
        });
      });
    }
  }
  
  

  /**
   * Saves the selected customizations for the current menu item.
   * @param menuItemId The ID of the menu item for which customizations are being saved.
   */
  saveCustomizations(menuItemId: number): void {
    console.log('Order Number:', this.orderNumber); 
    console.log('Menu Item ID:', menuItemId); 
    const ingredientIds = Object.keys(this.selectedIngredients)
      .filter(key => this.selectedIngredients[parseInt(key)].value)
      .join(',');
    console.log('Selected Ingredient IDs:', ingredientIds); 
    
    this.dataService.saveCustomizations(this.orderNumber, menuItemId, ingredientIds)
      .subscribe(() => {
        console.log('Customizations saved successfully');
        this.router.navigate(['/trevor-cart']);
      }, error => {
        console.error('Error saving customizations:', error);
      });
  }

  /**
   * Retrieves the image URL for the menu item.
   * @param id The ID of the menu item.
   * @returns The URL of the menu item image.
   */
  getMenuItemImage(id: number): string {
    return `assets/menu_item_images/item${id}.jpg`;
  }

  /**
   * Navigates back to the main menu.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }
  
  /**
   * Navigates to the cashier menu.
   */
  onSeeMenu(): void {
    this.router.navigate(['/trevor-cashier']);
  }

}
