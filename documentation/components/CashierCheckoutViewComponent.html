import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { DataService } from '../data.service';


/**
 * Component for displaying the checkout view for a cashier.
 */
@Component({
  selector: 'app-cashier-checkout-view',
  templateUrl: './cashier-checkout-view.component.html',
  styleUrl: './cashier-checkout-view.component.css'
})
export class CashierCheckoutViewComponent implements OnInit{

  /**
   * Constructs the cashier checkout view component.
   * @param router Provides the navigation and url manipulation capabilities.
   * @param dataService Provides access to data operations related to orders.
   */
  constructor(private router: Router, private dataService: DataService) { }

  /**
   * Stores the most recently completed order number.
   */
  orderNumber: number = 0;

  /**
   * Initializes the component by fetching the next order ID and setting the current order number.
   */
  ngOnInit(): void {
    this.dataService.getNextOrderId().subscribe(orderNumber => {
      this.orderNumber = orderNumber.next_order_id - 1;
    });
  }

  /**
   * Navigates to the cashier menu view.
   */
  goToCashierMenu(): void {
    console.log("Going back to cashier menu...");
    this.router.navigate(['/cashier-menu']);
  }

  /**
   * Navigates to the cashier cart view.
   */
  goToCashierCart(): void {
    console.log("Going back to cashier cart...");
    this.router.navigate(['/cashier-cart']);
  }

  /**
   * Array of text size options available for screen display.
   */
  textsizers = [
    { size: 'Normal', cssClass: 'text-normal' },
    { size: 'Large', cssClass: 'text-large' },
    { size: 'Extra Large', cssClass: 'text-extra-large' }
    ];
    
    /**
     * The currently selected text size for display.
     */
    selectedTextSize: string = 'text-normal';
    
    /**
     * Sets the text size based on the selected CSS class.
     * @param cssClass The CSS class that defines the text size.
     */
    setTextSize(cssClass: string): void {
      this.selectedTextSize = cssClass;
    }
    

}
