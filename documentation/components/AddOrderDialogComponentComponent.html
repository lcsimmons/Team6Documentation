import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators, FormArray } from '@angular/forms';
import { DataService } from '../data.service';
import { from } from 'rxjs';
import { concatMap, toArray } from 'rxjs/operators';
import { MatDialogRef } from '@angular/material/dialog';

/**
 * Component for adding a new order via a dialog component.
 */
@Component({
  selector: 'app-add-order-dialog-component',
  templateUrl: './add-order-dialog-component.component.html',
  styleUrls: ['./add-order-dialog-component.component.css']
})
export class AddOrderDialogComponentComponent implements OnInit {
  /**
   * Form group for creating a new order.
   */
  addOrderForm: FormGroup;
   /**
   * List of all menu items available for adding to the order.
   */
  allMenuItems: any[] = [];
  /**
   * Available status options for an order.
   */
  statusOptions = ['Cancelled', 'Fulfilled', 'In Progress'];

  /**
   * Constructor that sets up the form and injects necessary services.
   * @param formBuilder FormBuilder to create the form controls and groups.
   * @param dataService DataService to interact with backend data operations.
   * @param dialogRef Reference to the dialog that launched this component.
   */
  constructor(private formBuilder: FormBuilder, private dataService: DataService, public dialogRef: MatDialogRef<AddOrderDialogComponentComponent>) {
    this.addOrderForm = this.formBuilder.group({
      menuItems: this.formBuilder.array([]),
      togo: [false],
      employeeId: ['', Validators.required],
      status: ['', Validators.required]
    });
  }

  /**
   * On component initialization, fetches all available menu items from the data service.
   */
  ngOnInit(): void {
    this.dataService.getMenuItems().subscribe(menuItems => {
      this.allMenuItems = menuItems;
    });
  }

  /**
   * Accessor for the 'menuItems' form array, allowing dynamic addition of menu items to the order.
   * @returns The FormArray of menu items in the order form.
   */
  get menuItems(): FormArray {
    return this.addOrderForm.get('menuItems') as FormArray;
  }

  /**
   * Adds a new menu item control to the order form with a default selected item.
   */
  addMenuItem(): void {
    this.menuItems.push(this.formBuilder.control(this.allMenuItems[0].item_id));
  }

  /**
   * Handles form submission to create a new order. This method validates the form,
   * constructs the order data, calculates the total price, and uses the data service
   * to submit the new order to the backend.
   */
  onSubmit(): void {
    if (this.addOrderForm.valid) {
      const orderData = { ...this.addOrderForm.value };
      let totalPrice = 0;
      from(orderData.menuItems).pipe(
        concatMap((itemId: unknown) => this.dataService.getMenuItemByItemId((itemId as number))),
        toArray()
      ).subscribe(menuItems => {
        orderData.menuItems = menuItems.map(item => {
          totalPrice += item.price;
          return item.item_id; 
        });
  
        orderData.totalPrice = parseFloat(totalPrice.toFixed(2));
        orderData.date = new Date();
  
        this.dataService.getNextOrderId().subscribe(response => {
          orderData.orderId = response.next_order_id;
  
          // Add the order to the history
          this.dataService.addOrderHistory(
            orderData.orderId,
            orderData.menuItems.join(','),
            orderData.togo,
            orderData.totalPrice,
            orderData.date.toISOString(),
            orderData.employeeId,
            false // Assuming the order is not customized
          ).subscribe(response => {
            console.log(response);
            this.dialogRef.close();
          });
        });
      });
    }
  }
}
