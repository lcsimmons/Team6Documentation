import { Component, ElementRef, OnInit, ViewChild } from '@angular/core';
import { DataService } from '../data.service';
import { Router } from '@angular/router';
import { MatDialog } from '@angular/material/dialog';
import { EditOrderComponent } from '../edit-order/edit-order.component';
import { RemoveOrderItemsDialogComponent } from '../remove-order-items-dialog/remove-order-items-dialog.component';
import { AddOrderDialogComponentComponent } from '../add-order-dialog-component/add-order-dialog-component.component';

import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { AuthService } from '../auth.service';
import { tap } from 'rxjs/operators';

import { Subject, combineLatest } from 'rxjs';
import { takeUntil, catchError } from 'rxjs/operators';


import { MatSidenav } from '@angular/material/sidenav';
import { MatDatepickerInputEvent } from '@angular/material/datepicker';
import { forkJoin } from 'rxjs';
import { ViewCustomizationsComponent } from '../view-customizations/view-customizations.component';


/** Represent an Order */
interface Order {
  /** ID */
  id: number;
  /** Items in order */
  menu_items: any[];
  /** Is the order togo? */
  togo: boolean;
  /** Total price */
  total_price: number;
  /** Date of the order */
  date: string;
  /** Employee who took the order */
  employee: string;
  /** Status */
  status: string;
}

/**
 * Component to manage and interact with orders within a restaurant management system.
 */
@Component({
  selector: 'app-order-management',
  templateUrl: './order-management.component.html',
  styleUrl: './order-management.component.css'
})
export class OrderManagementComponent implements OnInit {

  /**
   * Reference to the side navigation component.
   */
  @ViewChild('datePickerInput') datePickerInput!: ElementRef;


  /**
   * Bionic variable to toggle the visual theme of the component.
   */
  Bionic = false;

  /**
   * Observable to determine if the device is a handset for responsive layout purposes.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
  );

  /**
   * Tracks the current page of orders being viewed
   */
  page: number = 1;

  /** Orders displayed on page */
  orders: any[] = [];
  /** Chart columns */
  displayedColumns: string[] = ['id', 'menu_items', 'togo', 'total_price', 'date', 'employee_id', 'status', 'actions'];


  /**
   * Constructor for the OrderManagementComponent.
   * @param dataService 
   * @param router 
   * @param dialog 
   * @param breakpointObserver 
   * @param authService 
   */
  constructor(private dataService: DataService, private router: Router, private dialog: MatDialog, private breakpointObserver: BreakpointObserver, private authService: AuthService) {
    this.datePickerInput = new ElementRef(null);
  }


  /** Show orders on initialization */
  ngOnInit(): void {
    this.loadOrders();
  }

  /**
   * Changes the displayed orders based on the selected date from the date picker.
   * @param event Datepicker event containing the new date.
   */
  onDateChange(event: MatDatepickerInputEvent<Date>) {
    const selectedDate = event.value;
    if (selectedDate) {
      const date = selectedDate.toISOString().split('T')[0];
      this.dataService.getOrdersByDate(date).subscribe((data: Order[]) => {
        this.orders = data;
      });
    }
  }

  /**
   * Navigates to the next page of orders.
   */
  nextPage(): void {
    this.page++;
    this.loadOrders();
  }

  /**
   * Navigates to the previous page of orders if not on the first page.
   */
  previousPage(): void {
    if (this.page === 1) {
      return;
    }
    this.page--;
    this.loadOrders();
  }

  /**
   * Logs out the current user.
   */
  logout(): void {
    this.authService.logout();
  }

  /**
   * Subject to unsubscribe from observables when the component is destroyed.
   */
  private unsubscribe$ = new Subject<void>();

  /**
   * Cleans up observables when the component is destroyed.
   */
  ngOnDestroy(): void {
    this.unsubscribe$.next();
    this.unsubscribe$.complete();
  }

  /**
   * Loads orders from the server based on the current page.
   */
  loadOrders(): void {
    this.dataService.get_last_orders(this.page).pipe(
      takeUntil(this.unsubscribe$),
      catchError(error => {
        // Handle error
        return [];
      })
    ).subscribe((data: Order[]) => {
      const ordersWithMenuItems$ = data.map(order => {
        let menuItemsArray: Observable<string>[] = [];
        if (Array.isArray(order.menu_items)) {
          menuItemsArray = order.menu_items.map((itemId: number) => this.getMenuItemName(itemId));
        } else if (typeof order.menu_items === 'string') {
          const parsedMenuItems = JSON.parse(order.menu_items);
          if (Array.isArray(parsedMenuItems)) {
            menuItemsArray = parsedMenuItems.map((itemId: number) => this.getMenuItemName(itemId));
          }
        }
        return combineLatest(menuItemsArray).pipe(
          map(menuItemNames => ({
            ...order,
            menu_items: menuItemNames
          }))
        );
      });

      combineLatest(ordersWithMenuItems$).pipe(
        takeUntil(this.unsubscribe$)
      ).subscribe(ordersWithMenuItems => {
        this.orders = ordersWithMenuItems;
      });
    });
  }
  

  /**
   * Fetches the name of a menu item by its ID.
   * @param itemId 
   * @returns 
   */
  getMenuItemName(itemId: number): Observable<string> {
    //console.log("Getting menu item for ID:", itemId);
    return this.dataService.getMenuItemNameById(itemId)
  }
  
  /**
   * Resets the date picker input field.
   */
  resetDatePicker() {
    if (this.datePickerInput && this.datePickerInput.nativeElement) {
      this.datePickerInput.nativeElement.value = '';
      this.loadOrders();
    }
  }

  /**
   * Creates an order based on the provided data.
   * @param order The order to create.
   */
  createOrder(order: any): void {
    this.dataService.createOrder(order).subscribe(() => {
      this.ngOnInit(); 
    });
  }

  /**
   * Updates the status of an order.
   * @param orderId The ID of the order to update.
   * @param event The event that triggered the update.
   */
  updateOrderStatus(orderId: string, event: Event): void {
    const target = event.target as HTMLSelectElement;
    const status = target.value;
  
    this.dataService.updateOrderStatus(orderId, status).subscribe(response => {
      console.log(response);
      this.loadOrders();
    });
  }

  /**
   * Opens the dialog to edit an order.
   * @param order The order data to be edited.
   */
  editOrder(order: any): void {
    const dialogRef = this.dialog.open(EditOrderComponent, {
      width: '700px',
      data: order
    });
  
    dialogRef.afterClosed().subscribe(() => {
      this.loadOrders();
    });
  }

  /**
   * Opens the dialog to remove items from an order.
   * @param order The order from which items are to be removed.
   */
  removeOrderItems(order: any): void {
    const dialogRef = this.dialog.open(RemoveOrderItemsDialogComponent, {
      width: '700px',
      data: order
    });
  
    dialogRef.afterClosed().subscribe(() => {
      this.loadOrders();
    });
  }

  /**
   * Opens the dialog to add a new order.
   */
  addOrder(): void {
    const dialogRef = this.dialog.open(AddOrderDialogComponentComponent, {
      width: '700px',
      data: {}
    });
  
    dialogRef.afterClosed().subscribe(() => {
      this.loadOrders();
    });
  }


  /**
   * Opens the dialog to view customizations for an order.
   * @param order The order to view customizations for.
   */
  viewCustomization(order: any): void {
    const dialogRef = this.dialog.open(ViewCustomizationsComponent, {
      width: '700px',
      data: order
    });
  
    dialogRef.afterClosed().subscribe(() => {
      this.loadOrders();
    });
  }


  /**
   * Navigates back to the employee login screen.
   */

  goHome(): void {
    this.router.navigate(['/employee-login']);
  }

  /**
   * Toggles the current visual theme of the component.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
  }
  
}
