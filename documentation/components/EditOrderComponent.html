import { Component, Inject, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, FormArray } from '@angular/forms';
import { MAT_DIALOG_DATA, MatDialogRef } from '@angular/material/dialog';
import { DataService } from '../data.service';
import { from } from 'rxjs';
import { concatMap, toArray } from 'rxjs/operators';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { AuthService } from '../auth.service';

/**
 * Component to handle the editing of an existing order.
 */
@Component({
  selector: 'app-edit-order',
  templateUrl: './edit-order.component.html',
  styleUrls: ['./edit-order.component.css']
})
export class EditOrderComponent implements OnInit {

  /**
   * Form group for the order form.
   */
  orderForm: FormGroup;
  /**
   * Array of all menu items available.
   */
  allMenuItems: any[] = [];

  /**
   * Observable to detect whether the device is a handset.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
  );

  /**
   * Constructs the component with injected dependencies and initializes the form group.
   * @param data The data injected into the component, containing details about the order.
   * @param fb FormBuilder service to construct the form group.
   * @param dialogRef Reference to the dialog that opened this component.
   * @param dataService Service to interact with data sources.
   * @param breakpointObserver Service to monitor screen size changes.
   * @param authService Service to handle authentication processes.
   */
  constructor(
    @Inject(MAT_DIALOG_DATA) public data: any,
    private fb: FormBuilder,
    public dialogRef: MatDialogRef<EditOrderComponent>,
    private dataService: DataService,
    private breakpointObserver: BreakpointObserver,
    private authService: AuthService
  ) {
    this.orderForm = this.fb.group({
      menuItems: this.fb.array(this.data.menuItems || []),
      togo: [data.togo],
      employeeId: [data.employee_id]
    });
  }

  /**
   * Fetches all menu items from the server to be used in the form on initialization.
   */
  ngOnInit() {
    this.dataService.getMenuItems().subscribe(menuItems => {
      this.allMenuItems = menuItems;
    });
  }

  /**
   * Logs out the current user.
   */
  logout(): void {
    this.authService.logout();
  }

  /**
   * Retrieves the 'menuItems' form array control.
   */
  get menuItems(): FormArray {
    return this.orderForm.get('menuItems') as FormArray;
  }

  /**
   * Adds a new menu item control to the 'menuItems' form array.
   */
  addMenuItem(): void {
    this.menuItems.push(this.fb.control(''));
  }

  /**
   * Submits the updated order form and processes it if valid.
   */
  onFormSubmit() {
    console.log(this.data);
    if (this.orderForm.valid) {
      const updatedOrder = { ...this.orderForm.value };
      let totalPrice = this.data.total_price;
      from(updatedOrder.menuItems).pipe(
        concatMap((itemId: unknown) => this.dataService.getMenuItemByItemId(itemId as number)),
        toArray()
      ).subscribe(menuItems => {
        updatedOrder.menuItems = menuItems.map(item => {
          totalPrice += item.price;
          return item.item_id; 
        });
  
        updatedOrder.totalPrice = parseFloat(totalPrice.toFixed(2));
  
        // Send the updated order to the backend
        this.dataService.updateOrder(this.data.id, updatedOrder)
          .subscribe(response => {
            console.log(response);
            this.dialogRef.close(updatedOrder);
          });
      });
    }
  }
}
