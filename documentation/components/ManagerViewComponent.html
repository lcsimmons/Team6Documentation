import { Component, ViewChild } from '@angular/core';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { Router } from '@angular/router';

/**
 * Component for the manager view that provides navigation and functionality toggles.
 */
@Component({
  selector: 'app-manager-view',
  templateUrl: './manager-view.component.html',
  styleUrls: ['./manager-view.component.css']
})
export class ManagerViewComponent {
  @ViewChild('drawer') drawer!: MatSidenav;
  toGo: boolean = false;

  /**
   * Observable to determine if the device is a handset for responsive layout purposes.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  constructor(private breakpointObserver: BreakpointObserver, private router: Router) { }

  /**
   * Toggles the to-go state for an order or feature.
   */
  toggleToGo(): void {
    this.toGo = !this.toGo;
  }

  /**
   * Navigates to the employee login page.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }
}
