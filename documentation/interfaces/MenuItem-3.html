import { Component, OnInit, ChangeDetectorRef } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { DataService } from '../data.service';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { Router } from '@angular/router';
import { MatDatepickerInputEvent } from '@angular/material/datepicker';
import { MatSnackBar } from '@angular/material/snack-bar';

/**
 * Interface for items on the menu, allowing type-safe usage of menu item data.
 */
export interface MenuItem {
  id: number;
  name: string; 
  price: number;
  selected?: boolean;
}

/**
 * Component for managing and editing the menu items within a restaurant management system.
 */
@Component({
  selector: 'app-manager-edit-menu',
  templateUrl: './manager-edit-menu.component.html',
  styleUrl: './manager-edit-menu.component.css'
})
export class ManagerEditMenuComponent {
  /** Flag for toggling the seasonal menu items display. */
  seasonal = false
  /** Stores today's date for comparison purposes. */
  TodayDate: string = '';
  /** Stores the end date for seasonal menu items. */
  SeasonalEndDate: string = '';
  /** Flag for theme toggling. */
  Bionic = false;
  /** Array of menu items fetched from the backend. */
  menuItems: MenuItem[] = [];
  /** Array of ingredients for the selected menu item. */
  menuItemIngredients: any[] = [];

  /** ID of the currently selected menu item. */
  selectedItemId: number | null = null;

  /** Form group for deleting menu items. */
  deleteForm: FormGroup;

  /** Form group for adding or editing menu items. */
  menuForm: FormGroup;

  /** Array of categories fetched for menu item categorization. */
  categories: any[] = []; 

  /** Array of ingredients fetched for menu item customization. */
  ingredients: any[] = []; 

  /** Flag to determine if an item is to go */
  toGo: boolean = false;

  /**
   * Observable to determine if the device is a handset for responsive layout purposes.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  constructor(private cdr: ChangeDetectorRef, private fb: FormBuilder, private dataService: DataService, private breakpointObserver: BreakpointObserver, private router: Router, private _snackbar: MatSnackBar) {
    this.menuForm = this.fb.group({
      name: ['', Validators.required],
      price: ['', [Validators.required, Validators.pattern(/^\d+\.?\d{0,2}$/)]],
      category_id: ['', Validators.required],
      ingredients: [[], Validators.required]
    });
    this.deleteForm = this.fb.group({
      menu_item_id: ['', Validators.required]
    });
    this.loadCategories();
    this.loadIngredients();
  }

  /**
   * Handles changes to the seasonal date picker, updating the seasonal end date.
   * @param event The date picker event.
   */
  onSeasonalDateChange(event: MatDatepickerInputEvent<Date>): void {
    this.seasonal = true;

    // console.log(event.value?.getDate());
    // console.log(event.value?.getMonth());
    // console.log(event.value?.getFullYear());
    
    this.SeasonalEndDate = event.value ? event.value.toISOString().slice(0, 10) : '';
  

    console.log("seasonal item: " + this.SeasonalEndDate);

    // console.log(event.value.toLocaleDateString('en-US'));

    // console.log("seasonal item");
  }

  /**
   * Loads categories from the server to populate the category selector.
   */
  loadCategories(): void {
    this.dataService.getCategories().subscribe(data => this.categories = data);
  }

  /**
   * Loads ingredients from the server for use in menu item compositions.
   */
  loadIngredients(): void {
    this.dataService.getIngredients().subscribe(data => this.ingredients = data);
  }


  ngOnInit(): void {
    let dateObj = new Date();
    this.TodayDate = dateObj.toISOString().slice(0, 10); // 2021-07-01

   
    this.dataService.getlastSeasonalItem().subscribe(data => {
      if(data != null){
        //console.log("last seasonal item: " + data.end_date);

        let SeasonalItemExpiryDate = new Date(data.end_date);
        let today = new Date(this.TodayDate);

        if(SeasonalItemExpiryDate > today) {
          console.log(SeasonalItemExpiryDate + " > " + today);
          console.log("dont delete seasonal item");

        } else if(SeasonalItemExpiryDate < today) {
          console.log(SeasonalItemExpiryDate + " < " + today);
          console.log("delete seasonal item");

          this.dataService.deleteSeasonalItem(data.end_date).subscribe({
            next: (res) => {
              console.log('Item deleted successfully', res);
              
            },
            error: (err) => {
              console.error('Error deleting item', err);
              
            }
          });
        } 
      }

    });

    this.loadMenuItems();
  }

  /**
   * Fetches and displays all menu items.
   */
  loadMenuItems(): void {
    this.dataService.getMenuItems().subscribe(data => {
      this.menuItems = data.map((item: { id: any; name: any; price: any; }) => ({
        id: item.id,
        name: item.name,
        price: item.price,
        selected: false
      }));
    });
  }

  /**
   * Toggles selection state of a menu item and loads its ingredients if selected.
   * @param item The menu item to toggle.
   */
  toggleMenuItem(item: MenuItem): void {
    if (item.id === this.selectedItemId) {
      this.selectedItemId = null;
      this.menuItemIngredients = [];
      item.selected = false;
    } else {
      this.menuItems.forEach(i => i.selected = i.id === item.id);
      this.selectedItemId = item.id;
      this.loadMenuItemIngredients(item.id);
    }
  }

  /**
   * Loads ingredients for a selected menu item.
   * @param itemId The ID of the menu item.
   */
  loadMenuItemIngredients(itemId: number): void {
    this.dataService.getMenuItem(itemId).subscribe(data => {
      this.menuItemIngredients = data.ingredients || []; 
    });
  }

  /**
   * Submits changes to the ingredients of a selected menu item.
   */
  submitChanges(): void {
    if (this.selectedItemId !== null) {
      const payload = {
        menu_item_id: this.selectedItemId,
        ingredients: this.menuItemIngredients
      };
      this.dataService.submitIngredientChanges(payload).subscribe(
        response => {
          console.log('Update successful', response);
        },
        error => {
          console.error('Update failed', error);
        }
      );
    } else {
      console.error('No menu item selected');
    }
  }

  /**
   * Submits a request to delete a menu item based on the form's validity.
   */
  submitDelete(): void {
    if (this.deleteForm.valid) {
      let itemId = this.deleteForm.get('menu_item_id')!.value;
      this.dataService.removeItemFromMenu(itemId).subscribe({
        next: (res) => {
         this.openSnackBar('Menu item deleted successfully');
          this.loadMenuItems(); 
        },
        error: (err) => {
          console.error('Deletion failed', err);
          this.openSnackBar('Error deleting menu item');
        }
      });
    } else {
      console.error('Form is not valid');
      this.openSnackBar('Invalid selection');
    }
  }

  

  /**
   * Adds or updates a seasonal menu item based on form validity and date checks.
   */
  submit(): void {
  
    if(this.seasonal){
      
      const item = {
        name: this.menuForm.get('name')?.value,
        price: this.menuForm.get('price')?.value,
        start_date: this.TodayDate,
        end_date: this.SeasonalEndDate
      };

      if (this.SeasonalEndDate <= this.TodayDate) {
        this.openSnackBar("Error: End date cannot be before current date.")
        return;
      }

      if (this.menuForm.valid) {
        this.dataService.addSeasonalMenuItem(item).subscribe({
          next: (res) => console.log('Seasonal menu item added successfully'),
          error: (err) => console.error('Error adding seasonal menu item', err)
        });
      }
      else {
        console.log("Invalid form")
      }

    }
    // else{
    if (this.menuForm.valid) {
        this.dataService.addMenuItem(this.menuForm.value).subscribe({
          next: (res) => this.openSnackBar('Menu item added successfully'),
          error: (err) => console.error('Error adding menu item', err)
        });
      }
      else {
        console.log("Invalid form")
      }
    // }
  }
  
  /**
   * Navigates back to the employee login screen.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }

  /**
   * Toggles the style theme and triggers change detection.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
    this.cdr.detectChanges(); 
  }


  /**
   * Displays a snack bar with a message.
   * @param message The message to display.
   */
  openSnackBar(message: string) {
    this._snackbar.open(message, "Done", {
      duration: 5000
    });
  }
  

}
