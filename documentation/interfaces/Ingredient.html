import { Component, EventEmitter, OnInit, Output, ViewChildren,QueryList, ElementRef } from '@angular/core';
import { ActivatedRoute, Router } from '@angular/router';
import { DataService } from '../data.service';
import { FormControl } from '@angular/forms';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';

/**
 * Represents an ingredient 
 */
interface Ingredient {
  /** ID of the ingredient in the database */
  id: number;
  /** Name of the item */
  name: string;
}

/**
 * Allow cashier to create custom orders
 */
@Component({
  selector: 'app-cashier-customize-item',
  templateUrl: './cashier-customize-item.component.html',
  styleUrls: ['./cashier-customize-item.component.css']
})

/**
 * Component for customizing menu items in the cashier interface.
 */
export class CashierCustomizeItemComponent implements OnInit {

  /**
   * ViewChildren query list to access DOM elements with the reference 'readContent'.
   */
  @ViewChildren('readContent') readContents!: QueryList<ElementRef>;

   /**
   * The menu item being customized.
   */
  menuItem: any;

  /**
   * The list of available ingredients for customization.
   */
  ingredients: Ingredient[] = [];

  /**
   * Object to track selected ingredients using their IDs as keys.
   */
  selectedIngredients: { [key: number]: FormControl } = {};

  /**
   * The order number associated with the customization process.
   */
  orderNumber: number = 0;

  /**
   * The list of available customization choices for the menu item.
   */
  customizationChoices: { id: number; name: string; price: number }[] = []; // Include price in the customization choices

  /**
   * The list of customizations already applied to the menu item.
   */
  customizations: any[] = [];

  /**
   * Object to track selected customizations using their IDs as keys.
   */
  selectedCustomizations: { [key: number]: FormControl } = {};

  /**
 * EventEmitter for notifying changes in the total price.
 */
  @Output() totalPriceChanged: EventEmitter<number> = new EventEmitter<number>(); // Event emitter for total price

  /**
   * Observable to determine if the device is a handset.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );
  
  /**
   * Flag indicating whether the accessibility toolbar container is visible.
   */
  isAccessibilityToolbarContainerVisible: boolean = true;

  /**
   * Toggle if accessibility bar is shown
   */
  toggleAccessibilityToolbarContainer(): void {
    this.isAccessibilityToolbarContainerVisible = !this.isAccessibilityToolbarContainerVisible;
    console.log('Toolbar visibility:', this.isAccessibilityToolbarContainerVisible);
  }
  

   /**
   * Array of text size options available for screen display.
   */
  textsizers = [
    { size: 'Normal', cssClass: 'text-normal' },
    { size: 'Large', cssClass: 'text-large' },
    { size: 'Extra Large', cssClass: 'text-extra-large' }
    ];
    
    /**
     * The currently selected text size for display.
     */
    selectedTextSize: string = 'text-normal';
    
    /**
     * Sets the text size based on the selected CSS class.
     * @param cssClass The CSS class that defines the text size.
     */
    setTextSize(cssClass: string): void {
      this.selectedTextSize = cssClass;
    }
    

  /* end accessibility toolbar code */

  /**
   * Description of the weather.
   */
  weatherDescription: string = '';

  /**
   * Current temperature.
   */
  temperature: number = 0;

  /**
   * Feels-like temperature.
   */
  feelsLikeTemperature: number = 0;

  /**
   * Weather icon representation.
   */
  weathericon: string = '01d.png';


  /**
   * Constructs the component with necessary dependencies.
   * @param route ActivatedRoute for accessing route parameters.
   * @param dataService DataService for fetching and submitting data related to menu items.
   * @param router Router for navigation.
   * @param breakpointObserver BreakpointObserver for reacting to media query changes.
   */
  constructor(
    private route: ActivatedRoute,
    private dataService: DataService,

    private router: Router,
    private breakpointObserver: BreakpointObserver
  ) {
    this.customizationChoices = [
      { id: 1, name: 'Crispy Bacon', price: 0.81 },
      { id: 2, name: 'American Cheese', price: 0.65 },
      { id: 4, name: 'Lettuce', price: 0.36 },
      { id: 5, name: 'Pickles', price: 0.18 },
      { id: 6, name: 'Tomatoes', price: 0.78 },
      { id: 7, name: 'Onions', price: 0.10 },
      { id: 10, name: 'Caramelized Onions', price: 0.40 },
      { id: 11, name: 'American Swiss Cheese', price: 0.86 }
    ];

    // Initialize selectedCustomizations with all customizations from customizationChoices
    this.customizationChoices.forEach(choice => {
      // For ingredients, set initial value to true
      this.selectedIngredients[choice.id] = new FormControl(false);
      // For extra customizations, set initial value to false
      this.selectedCustomizations[choice.id] = new FormControl(false);
    });
  }   
    

  /** Initilize component */
  ngOnInit(): void {
    this.dataService.getNextOrderId().subscribe((response: any) => {
      this.orderNumber = response.next_order_id;
    });

    const itemId = this.route.snapshot.paramMap.get('id');
    if (itemId !== null) {
      this.dataService.getMenuItem(+itemId).subscribe((menuItem: any) => {
        this.menuItem = menuItem;
        this.ingredients = menuItem.ingredients;
        this.selectedIngredients = {};
        this.ingredients.forEach(ingredient => {
          this.selectedIngredients[ingredient.id] = new FormControl(true);
        });

        this.loadCustomizations();
      });
    } else {
      console.error('ID parameter is null');
    }
  }

  /** Text to speech menu */
  speakText(): void {
    let fullText = '';
    this.readContents.forEach((elementRef) => {
      fullText += this.getTextFromElement(elementRef.nativeElement) + ' ';
    });

    const utterance = new SpeechSynthesisUtterance(fullText);
    window.speechSynthesis.speak(utterance);
  }

  /**
   * Get text from element
   * @param element DOM element
   * @returns text
   */
  private getTextFromElement(element: HTMLElement): string {
    let text = "";
    const children = Array.from(element.childNodes);
    children.forEach((node) => {
      if (node.nodeType === Node.TEXT_NODE) {
        text += node.textContent?.trim() + " ";
        text += ". ";
      } else if (node.nodeType === Node.ELEMENT_NODE) {
        text += this.getTextFromElement(node as HTMLElement);
      }
    });
    console.log('Text: ', text);
    return text;
  }

  /**
   * Save customizations
   */
  loadCustomizations(): void {
    console.log('Loading customizations for order ID:', this.orderNumber);
    console.log('Menu item ID:', this.menuItem?.id); // Add null check with '?'
    
    if (this.orderNumber !== 0 && this.menuItem !== undefined) {
      this.dataService.getCustomizations(this.orderNumber, this.menuItem.id).subscribe((customizations: any[]) => {
        console.log('Customizations loaded:', customizations);
        customizations.forEach(customization => {
          const ingredientId = customization;
          if (this.selectedIngredients[ingredientId]) {
            this.selectedIngredients[ingredientId].setValue(true);
          }
          // Check for extra customizations and set their form control values
          if (this.selectedCustomizations[ingredientId]) {
            this.selectedCustomizations[ingredientId].setValue(true);
          }
        });
      });
    }
  }
  


  /**
   * Save customizations
   * @param menuItemId ID of item
   */
  saveCustomizations(menuItemId: number): void {
    console.log('Order Number:', this.orderNumber);
    console.log('Menu Item ID:', menuItemId);
    
    // Calculate the total price of selected customizations
    const totalPrice = this.calculateTotalPrice();

    // Emit the total price
    this.totalPriceChanged.emit(totalPrice);

    // Convert array of strings to a single comma-separated string for selected ingredients
    const selectedIngredients: string = Object.keys(this.selectedIngredients)
      .filter(key => this.selectedIngredients[parseInt(key)].value)
      .map(key => parseInt(key).toString()) // Convert numbers to strings
      .join(','); // Join the array elements with commas

    // Convert array of strings to a single comma-separated string for selected customizations
    const selectedExtraCustomizations: string = Object.keys(this.selectedCustomizations)
      .filter(key => this.selectedCustomizations[parseInt(key)].value)
      .map(key => parseInt(key).toString()) // Convert numbers to strings
      .join(','); // Join the array elements with commas

    console.log('Selected Ingredients:', selectedIngredients);
    console.log('Selected Extra Customizations:', selectedExtraCustomizations);
    console.log('Total Price:', totalPrice);

    // Save the customizations to the database
    this.dataService.saveCustomizations(this.orderNumber, menuItemId, selectedIngredients + ',' + selectedExtraCustomizations)
      .subscribe(() => {
        console.log('Customizations saved successfully');
        this.router.navigate(['/cashier-cart']);
      }, error => {
        console.error('Error saving customizations:', error);
      });
  }


  /**
   * Calculate total price
   * @returns Total Price
   */
  calculateTotalPrice(): number {
    let totalPrice = 0;
  
    Object.keys(this.selectedCustomizations).forEach(key => {
      const customizationId = parseInt(key);
      if (this.selectedCustomizations[customizationId].value) {
        const customization = this.customizationChoices.find(c => c.id === customizationId);
        if (customization) {
          // Add 50 cents for each extra customization
          totalPrice += customization.price + 0.50;
        }
      }
    });
  
    return totalPrice;
  }
  

  getMenuItemImage(id: number): string {
    return `assets/menu_item_images/item${id}.jpg`;
  }

  /** Navigate to home page */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }
  /** Navigate to menu page */
  onSeeMenu(): void {
    this.router.navigate(['/cashier-menu']);
  }

  handleImageError(event: any) {
    event.target.src = 'assets/REVSLogoUpdated.jpg'; // Replace with path to your placeholder image
  }

}
