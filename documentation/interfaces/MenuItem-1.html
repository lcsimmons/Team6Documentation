import { Component, OnInit, ViewChild, ViewChildren, QueryList, ElementRef } from '@angular/core';
import { DataService } from '../data.service';
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';
import { MatSnackBar } from '@angular/material/snack-bar';

/**
 * Interface to define menu items
 */
export interface MenuItem {
  /** Numeric ID of the item */
  id: number;
  /** Name of the menu item */
  name: string; 
  /** Price of the item */
  price: number;
  /** How many are being ordered */
  quantity: number;
}

/**
 * Component for the cashier to manage menu selections and operations.
 */
@Component({
  selector: 'app-cashier-menu',
  templateUrl: './cashier-menu.component.html',
  styleUrls: ['./cashier-menu.component.css'],
})
export class CashierMenuComponent implements OnInit {
  @ViewChild('drawer') drawer!: MatSidenav;
  // -----
  // @ViewChild('readContent', { static: true }) readContent!: ElementRef;
  @ViewChildren('readContent') readContents!: QueryList<ElementRef>;


  menuItems: MenuItem[] = [];
  selectedMenuItem?: MenuItem;
  selectedCategory?: string;


  categories: string[] = ['Burger', 'Sandwich', 'Value Meal', 'Basket', 'Side', 'Shakes & More', 'Beverage', 'Extras', 'Sauces', 'Seasonal'];

  /**
   * Observable to determine if the device is a handset.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  /* accesibility toolbar code here */
  isAccessibilityToolbarContainerVisible: boolean = true;

  /**
   * Toggles the visibility of the accessibility toolbar.
   */
  toggleAccessibilityToolbarContainer(): void {
    this.isAccessibilityToolbarContainerVisible = !this.isAccessibilityToolbarContainerVisible;
    console.log('Toolbar visibility:', this.isAccessibilityToolbarContainerVisible);
  }
  
   /**
   * Array of text size options available for screen display.
   */
   textsizers = [
    { size: 'Normal', cssClass: 'text-normal' },
    { size: 'Large', cssClass: 'text-large' },
    { size: 'Extra Large', cssClass: 'text-extra-large' }
    ];
    
    /**
     * The currently selected text size for display.
     */
    selectedTextSize: string = 'text-normal';
    
    /**
     * Sets the text size based on the selected CSS class.
     * @param cssClass The CSS class that defines the text size.
     */
    setTextSize(cssClass: string): void {
      this.selectedTextSize = cssClass;
    }
    

  /* end accessibility toolbar code */



  constructor(private dataService: DataService, private router: Router, private breakpointObserver: BreakpointObserver, private _snackBar: MatSnackBar) { }

  /**
   * Initializes the component by selecting the 'Burger' category by default.
   */
  ngOnInit(): void {
    this.onSelectCategory("Burger");
  }

  /**
   * Uses the Web Speech API to speak the text content of selected elements.
   */
  speakText(): void {
    let fullText = '';
    this.readContents.forEach((elementRef) => {
      fullText += this.getTextFromElement(elementRef.nativeElement) + ' ';
    });

    const utterance = new SpeechSynthesisUtterance(fullText);
    window.speechSynthesis.speak(utterance);
  }

  /**
   * Recursively retrieves text content from an HTML element, including its children.
   * @param element The HTMLElement from which to extract text.
   * @returns Concatenated text of the element and its children.
   */
  private getTextFromElement(element: HTMLElement): string {
    let text = "";
    const children = Array.from(element.childNodes);
    children.forEach((node) => {
      if (node.nodeType === Node.TEXT_NODE) {
        text += node.textContent?.trim() + " ";
        text += ". ";
      } else if (node.nodeType === Node.ELEMENT_NODE) {
        text += this.getTextFromElement(node as HTMLElement);
      }
    });
    console.log('Text: ', text);
    return text;
  }

  /**
   * Updates the quantity and total price for a menu item.
   * @param item The menu item to update.
   */
  updateQuantity(item: any): void {
    item.totalPrice = item.price * item.quantity;
  }

  /**
   * Fetches menu items for a selected category.
   * @param category The category to fetch menu items for.
   */
  onSelectCategory(category: string): void {
    this.selectedCategory = category;
    this.dataService.getMenuItemsByCategory(category).subscribe(
      (items: any[]) => {
        if (Array.isArray(items)) {
          this.menuItems = items.map(item => ({
            id: item.id,
            name: item.name,
            price: item.price,
            quantity: 1
          }));
        } else {
          console.error('Data received from API is not an array:', items);
        
        }
      },
      error => {
        console.error('Error fetching menu items:', error);
      }
    );
  }

  /**
   * Selects or deselects a menu item.
   * @param item The menu item to select or deselect.
   */
  onSelectMenuItem(item: MenuItem): void {
    if (this.selectedMenuItem && this.selectedMenuItem.id === item.id) {
      this.selectedMenuItem = undefined;
    } else {
      this.selectedMenuItem = item;
    }
  }

  /**
   * Checks if a category is currently selected.
   * @param category The category to check.
   * @returns True if the category is selected, false otherwise.
   */
  isCategorySelected(category: string): boolean {
    return this.selectedCategory == category;
  }

  /**
   * Checks if a menu item is selected.
   * @param item The item to check.
   * @returns True if the item is selected, false otherwise.
   */
  isItemSelected(item: MenuItem): boolean {
    return this.selectedMenuItem?.name === item.name;
  }

  /**
   * Navigates to the cashier cart view.
   */
  onSeeCart(): void {
    this.drawer.close();
    this.router.navigate(['/cashier-cart']);
  }

  /**
   * Adds a selected menu item to the cart.
   */
  addToCart(): void {
    if (this.selectedMenuItem && this.selectedMenuItem.quantity > 0) {
      for (let i = 0; i < this.selectedMenuItem.quantity; i++) {
        this.dataService.addToCart(this.selectedMenuItem);
      }
      this.selectedMenuItem = undefined;
      this.openSnackBar("Item added to cart.")
    }
  }

  /**
   * Logs out the user and navigates to the login screen.
   */
  goHome(): void {
    this.dataService.clearCart();
    this.router.navigate(['/employee-login']);
  }

  /**
   * Converts a selected menu item into a combo by adding fries and a drink.
   */
  makeCombo(): void {
    if (this.selectedMenuItem) {
      // Fetch fries and 20 oz fountain drink items from data
      const friesItem: MenuItem = { id: 14, name: 'French Fries', price: 1.99, quantity: 1 };
      const drinkItem: MenuItem = { id: 25, name: '20 oz. Fountain Drink', price: 1.99, quantity: 1 };

      // Add selected item, fries, and drink to the cart
      this.dataService.addToCart(this.selectedMenuItem);
      this.dataService.addToCart(friesItem);
      this.dataService.addToCart(drinkItem);

      this.openSnackBar("Item updated as a combo.")
    }
  }

  /**
   * Displays a snack bar with a message.
   * @param message The message to display.
   */
  openSnackBar(message: string) {
    this._snackBar.open(message, "Done", {
      duration: 5000
    });
  }
}
