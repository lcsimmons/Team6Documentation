  /**
 * Constructs an instance of the CashierCartComponent.
 * @param dataService The data service used to retrieve cart items.
 * @param router The router service for navigating between routes.
 * @param breakpointObserver The breakpoint observer service for monitoring screen size changes.
 */
  constructor(private dataService: DataService, private router: Router, private breakpointObserver: BreakpointObserver) { }

  /**
 * Lifecycle hook that is called after Angular has initialized all data-bound properties of a directive.
 * It is a good place to perform initialization logic.
 */
  ngOnInit(): void {
    this.dataService.getCartItems().subscribe(items => {
      this.cartItems = items;
    });
  }

  /**
 * Speaks the text content of elements stored in `readContents` array.
 */
  speakText(): void {
    let fullText = '';
    this.readContents.forEach((elementRef) => {
      fullText += this.getTextFromElement(elementRef.nativeElement) + ' ';
    });

    const utterance = new SpeechSynthesisUtterance(fullText);
    window.speechSynthesis.speak(utterance);
  }

  /**
 * Recursively retrieves text content from a given HTML element.
 * @param element The HTML element to extract text content from.
 * @returns The text content of the element and its children.
 */
  private getTextFromElement(element: HTMLElement): string {
    let text = "";
    const children = Array.from(element.childNodes);
    children.forEach((node) => {
      if (node.nodeType === Node.TEXT_NODE) {
        text += node.textContent?.trim() + " ";
        text += ". ";
      } else if (node.nodeType === Node.ELEMENT_NODE) {
        text += this.getTextFromElement(node as HTMLElement);
      }
    });
    console.log('Text: ', text);
    return text;
  }

  /**
 * Toggles the value of `toGo` between true and false.
 */
  toggleToGo(): void {
    this.toGo = !this.toGo;
  }

  /**
 * Navigates to the cashier-customize-item route with the specified item ID.
 * @param itemId The ID of the item to be edited.
 */
  editItem(itemId: number): void {
    console.log('Navigating to:', ['/cashier-customize-item', itemId]);
    this.router.navigate(['/cashier-customize-item', itemId]);
  }

  /**
 * Processes checkout for the items in the cart.
 * Adds order history, clears the cart, and navigates to the cashier-checkout-view route.
 */
  checkout(): void {
    console.log("Checking out...");

    console.log("Items in cart:");
  
    this.dataService.getNextOrderId().pipe(
      concatMap((orderNumber: any) => {
        this.orderNumber = orderNumber.next_order_id;
  
        let observables = this.cartItems.map(item => {
          return this.dataService.getItemIdByItemName(item.name);
        });
        return forkJoin(observables);
      }),
      concatMap((itemIds: ItemId[]) => {
        this.itemListString = itemIds.map(item => item.itemId).join(", ");
  
        const totalPrice = this.dataService.calculateTotalCost();
        const time = new Date().toISOString();
        const employeeId = 1;
        const isCustomized = false;
  
        return this.dataService.addOrderHistory(this.orderNumber, this.itemListString, this.toGo, totalPrice, time, employeeId, isCustomized);
      })
    ).subscribe(() => {
      this.dataService.clearCart();
      this.cartItems = [];
      this.router.navigate(['/cashier-checkout-view']);
    });
  }

  /**
 * Removes an item from the cart by its index.
 * @param index The index of the item to be removed.
 */
  removeItem(index: number): void {
    if (index > -1 && index < this.cartItems.length) {
      const item = this.cartItems[index];
      this.cartItems.splice(index, 1);
      this.dataService.removeItemFromCart(item);
    }
  }

  /**
 * Calculates the total price of items in the cart including taxes.
 * @returns The total price including taxes.
 */
  calculateTotalPrice(): number {
    const subtotal = this.cartItems.reduce((total, item) => total + item.price, 0);
    const taxes = this.calculateTaxes();
    return subtotal + taxes;
  }

  /**
 * Calculates the subtotal of items in the cart.
 * @returns The subtotal of items.
 */
  calculateSubtotal(): number {
    return this.cartItems.reduce((total, item) => total + item.price, 0);
  }


  /**
 * Calculates the total taxes based on the subtotal and a fixed tax rate.
 * @returns The total taxes.
 */
  calculateTaxes(): number {
    const taxRate = 0.0625;
    return this.calculateSubtotal() * taxRate;
  }

  /**
 * Navigates to the employee-login route.

  goHome(): void {
    this.router.navigate(['/employee-login']);
  }
  
  /**
 * Navigates to the cashier-menu route.

  onSeeMenu(): void {
    this.router.navigate(['/cashier-menu']);
  }

}
