import { Component, OnInit, ViewChild } from '@angular/core';
import { DataService } from '../data.service';
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { map, shareReplay } from 'rxjs/operators';

/**
 * Interface for menu item objects
 */
export interface MenuItem {
  /** Numeric ID of the item */
  id: number;
  /** Name of the menu item */
  name: string; 
  /** Price of the item */
  price: number;
}

/**
 * Acessible component for managing cashier functionalities such as selecting items,
 * adding to cart, and viewing the cart.
 */
@Component({
  selector: 'app-trevor-cashier',
  templateUrl: './trevor-cashier.component.html',
  styleUrl: './trevor-cashier.component.css'
})
export class TrevorCashierComponent {
  /**
   * Reference to the side navigation component.
   */
  @ViewChild('drawer') drawer!: MatSidenav;
  /**
   *  Array of menu items to display in the cashier view.
   */
  menuItems: MenuItem[] = [];
  /**
   * The currently selected menu item for customization or addition to cart.
   */
  selectedMenuItem?: MenuItem;
  /**
   * The currently selected category of menu items.
   */
  selectedCategory?: string;


  /**
   * Array of categories for menu items.
   */
  categories: string[] = ['Burger', 'Sandwich', 'Value Meal', 'Basket', 'Side', 'Shakes & More', 'Beverage', 'Extras', 'Sauces'];

  /**
   * Observable that emits true if the screen is a handset device, false otherwise.
   */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  /**
   * Accessibility toolbar container visibility status.
   */
  isAccessibilityToolbarContainerVisible: boolean = true;

  /**
   * Toggles the visibility of the accessibility toolbar container.
   */
  toggleAccessibilityToolbarContainer(): void {
    this.isAccessibilityToolbarContainerVisible = !this.isAccessibilityToolbarContainerVisible;
    console.log('Toolbar visibility:', this.isAccessibilityToolbarContainerVisible);
  }
  
  /**
   * Increases the font size of the page.
    */
  textsizers = [
    { size: 'Regular'},
    { size: 'Large'},
    { size: 'Extra Large'},
  ]

  /* end accessibility toolbar code */


  /**
   * Weather data for College Station, TX.
   */
  weatherDescription: string = '';
  /**
   * Temperature in Fahrenheit for College Station, TX.
   */
  temperature: number = 0;
  /**
   * Feels-like temperature in Fahrenheit for College Station, TX.
   */
  feelsLikeTemperature: number = 0;
  /**
   * Weather icon for College Station, TX.
   */
  weathericon: string = '01d.png';

  /**
   * Constructs the cashier component with necessary dependencies.
   * @param dataService DataService to fetch and manage item data.
   * @param router Router for navigation.
   * @param breakpointObserver BreakpointObserver to respond to media query changes.
   */
  constructor(private dataService: DataService, private router: Router, private breakpointObserver: BreakpointObserver) { }

  /**
   * Initializes the cashier component by fetching menu items and weather data.
   * Also loads the Google Translate script.
   * @returns void
   */
  ngOnInit(): void {
    this.onSelectCategory("Burger");
    this.loadGoogleTranslate();
    this.callWeatherApi();
  }

  /**
   * Calls the OpenWeatherMap API to fetch weather data for College Station, TX.
   */
  callWeatherApi(): void {
    const lat = 30.6125364
    const lon = -96.3433166
    const api_key = 'dcd3c687ea0ab81cb49468a99ae0ddb4'
    
    
    this.dataService.getWeather(lat, lon, api_key).subscribe((weather: any) => {
      this.weatherDescription = weather.current.weather[0].description;
      this.temperature = Math.round((weather.current.temp - 273.15) * (9 / 5) + 32);
      // this.feelsLikeTemperature = Math.round((weather.current.feels_like - 273.15) * (9 / 5) + 32);
      if(weather.current.weather[0].icon != null){
        if(weather.current.weather[0].icon == '01d'){
          this.weathericon = '01d.png';

        if(weather.current.weather[0].icon == '01n'){
          this.weathericon = '01n.png';
        }
        if(weather.current.weather[0].icon == '02d'){
          this.weathericon = '02d.png';
        }
        if(weather.current.weather[0].icon == '02n'){
          this.weathericon = '02n.png';
        }
        if(weather.current.weather[0].icon == '03d' || weather.current.weather[0].icon == '03n'){
          this.weathericon = '03d.png';
        }
        if(weather.current.weather[0].icon == '04d' || weather.current.weather[0].icon == '04n'){
          this.weathericon = '04d.png';

          this.weathericon = '09d.png';

          this.weathericon = '10d.png';

        if(weather.current.weather[0].icon == '10n'){

  }

  /**
   * Loads the Google Translate script to enable translation of the page.
   * @returns void
   */
  loadGoogleTranslate(): void {
    const existingScript = document.getElementById('google-translate-script');
    if (!existingScript) {
      const script = document.createElement('script');
      script.id = 'google-translate-script';
      script.type = 'text/javascript';
      script.src = '//translate.google.com/translate_a/element.js?cb=googleTranslateElementInit';
      document.head.appendChild(script);
    }
      (window as any).googleTranslateElementInit = this.initGoogleTranslate;
  }

  /**
   * Initializes the Google Translate element on the page.
   * @returns void
   */
  initGoogleTranslate(): void {
    new window.google.translate.TranslateElement({
      pageLanguage: 'en',
      layout: window.google.translate.TranslateElement.InlineLayout.SIMPLE
    }, 'google_translate_element');
  }

  /**
   * Fetches and displays menu items based on selected category.
   * @param category The category of items to display.
   */
  onSelectCategory(category: string): void {
    this.selectedCategory = category;
    this.dataService.getMenuItemsByCategory(category).subscribe(
      (items: any[]) => {
        if (Array.isArray(items)) {
          this.menuItems = items.map(item => ({
            id: item.id,
            name: item.name,
            price: item.price
          }));
        } else {
          console.error('Data received from API is not an array:', items);
        
        }
      },
      error => {
        console.error('Error fetching menu items:', error);
      }
    );
  }

  /**
   * Selects a menu item for further actions like customization or addition to cart.
   * @param item The menu item selected.
   */
  onSelectMenuItem(item: MenuItem): void {
    if (this.selectedMenuItem && this.selectedMenuItem.id === item.id) {
      this.selectedMenuItem = undefined;
    } else {
      this.selectedMenuItem = item;
    }
  }

  /**
   * Checks if a category is currently selected.
   * @param category The category to check.
   * @returns True if the category is selected, false otherwise.
   */
  isCategorySelected(category: string): boolean {
    return this.selectedCategory == category;
  }

  /**
   * Checks if a menu item is currently selected.
   * @param item The item to check.
   * @returns True if the item is selected, false otherwise.
   */
  isItemSelected(item: MenuItem): boolean {
    return this.selectedMenuItem?.name === item.name;
  }

  /**
   * Opens the cart view.
   */
  onSeeCart(): void {
    this.drawer.close();
    this.router.navigate(['/trevor-cart']);
  }

  /**
   * Adds the selected menu item to the cart.
   */
  addToCart(): void {
    if (this.selectedMenuItem) {
      this.dataService.addToCart(this.selectedMenuItem);
      this.selectedMenuItem = undefined;
    }
  }

  /**
   * Navigates back to the employee login screen.
   */
  goHome(): void {
    this.dataService.clearCart();
    this.router.navigate(['/employee-login']);
  }

  /**
   * Combines a selected item with fries and a drink to create a combo and adds it to the cart.
   */
  makeCombo(): void {
    if (this.selectedMenuItem) {
      // Fetch fries and 20 oz fountain drink items from data
      const friesItem: MenuItem = { id: 14, name: 'French Fries', price: 1.99 };
      const drinkItem: MenuItem = { id: 25, name: '20 oz. Fountain Drink', price: 1.99 };

      // Add selected item, fries, and drink to the cart
      this.dataService.addToCart(this.selectedMenuItem);
      this.dataService.addToCart(friesItem);
      this.dataService.addToCart(drinkItem);
    }
  }

  /**
   * Handles image loading errors by setting a default image.
   * @param event The error event from image loading failure.
   */
  handleImageError(event: any) {
    event.target.src = 'assets/REVSLogoUpdated.jpg'; // Replace with path to your placeholder image
  }
  
}
