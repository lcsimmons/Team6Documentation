import { Component, OnInit, ViewChild } from '@angular/core';
import { DataService } from '../data.service';
import { Router } from '@angular/router';
import { MatSidenav } from '@angular/material/sidenav';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';
import { Observable } from 'rxjs';
import { concatMap, map, shareReplay } from 'rxjs/operators';
import { forkJoin } from 'rxjs';
import { FormBuilder, FormGroup } from '@angular/forms';
import { MatSnackBar } from '@angular/material/snack-bar';
import { DatePipe } from '@angular/common';

/**
 * Interface to define menu items
 */
export interface MenuItem {
 /** Numeric ID of the item */
 id: number;
 /** Name of the menu item */
 name: string; 
 /** Price of the item */
 price: number;
}

/**
 * Component to display and manage popularity metrics for menu items over selected time frames.
 */
@Component({
  selector: 'app-menu-popularity',
  templateUrl: './menu-popularity.component.html',
  styleUrls: ['./menu-popularity.component.css'],
  providers: [DatePipe]
})

/**
 * Component for displaying the popularity of menu items within a specified date range.
 */
export class MenuPopularityComponent implements OnInit {

  /** Flag indicating the current visual theme. */
  Bionic = false; 
  /** Reference to the MatSidenav element. */
  @ViewChild('drawer') drawer!: MatSidenav;
  /** Form group for selecting date range. */
  dateForm!: FormGroup;
  
  /** Number of top menu items to display. */
  topN: number = 10;
  
  /** Start date for the date range. */
  date1: Date = new Date();

  /** End date for the date range. */
  date2: Date = new Date();

  /** String representation of the start date. */
  date1String: string = this.date1.toISOString().split('T')[0];

  /** String representation of the end date. */
  date2String: string = this.date2.toISOString().split('T')[0];

  /** Displayed columns in the top menu items table. */
  displayedColumns: string[] = ['id', 'name', 'price'];
  /** Top menu items based on popularity. */
  topMenuItems: MenuItem[] = [];

  /**
 * Constructs an instance of the MenuPopularityComponent.
 * @param fb The FormBuilder service for creating form groups.
 * @param router The router service for navigating between routes.
 * @param dataService The data service for fetching menu items and related data.
 * @param breakpointObserver The breakpoint observer service for monitoring screen size changes.
 * @param _snackBar The snack bar service for displaying notifications.
 * @param datePipe The date pipe service for formatting dates.
 */
  constructor(private fb: FormBuilder, private router: Router, private dataService: DataService, private breakpointObserver: BreakpointObserver, private _snackBar: MatSnackBar, private datePipe: DatePipe) {
      this.dateForm = this.fb.group({
      startDate: [''],
      endDate: [''],
      range: ['']
    });
  }

  /** Observable representing whether the device is a handset. */
  isHandset$: Observable<boolean> = this.breakpointObserver.observe(Breakpoints.Handset)
    .pipe(
      map(result => result.matches),
      shareReplay()
    );

  /**
 * Lifecycle hook that is called after Angular has initialized all data-bound properties of a directive.
 * It is a good place to perform initialization logic.
 */
  ngOnInit(): void {
  }

  /**
   * Updates the date range for the popularity report based on the selected range option.
   * @param event HTML event containing the selected range.
   */
  updateDateRange(event: Event) {
    const target = event.target as HTMLSelectElement;
    const range = target.value;

    let actualStartDate: Date;
    let actualEndDate: Date;
    const now = new Date();

    switch (range) {
        case 'month':
            actualStartDate = new Date(now.getFullYear(), now.getMonth() - 1, now.getDate());
            actualEndDate = now;
            break;
        case 'year':
            actualStartDate = new Date(now.getFullYear() - 1, now.getMonth(), now.getDate());
            actualEndDate = now;
            break;
        case '2weeks':
            actualStartDate = new Date(now.getFullYear(), now.getMonth(), now.getDate() - 14);
            actualEndDate = now;
            break;
        default:
            actualStartDate = now;
            actualEndDate = now;
    }

    this.date1 = actualStartDate;
    this.date2 = actualEndDate;

    this.date1String = this.date1.toISOString().split('T')[0];
    this.date2String = this.date2.toISOString().split('T')[0];

    this.updateTable(actualStartDate, actualEndDate, this.topN);
}

/**
 * Updates the table with the most popular menu items between selected dates.
 * @param startDate The starting date for the report.
 * @param endDate The ending date for the report.
 * @param topN Number of top items to retrieve.
 */
updateTable(startDate: Date, endDate: Date, topN: number): void {
  const start = this.datePipe.transform(startDate, 'yyyy-MM-dd') || '';
  const end = this.datePipe.transform(endDate, 'yyyy-MM-dd') || '';

  this.dataService.getTopMenuItems(start, end, topN).pipe(
      concatMap((data: any) => {
          // Create an array of Observables, one for each menu item
          const observables = data.top_menu_items.map((item: any) => {
              return this.dataService.getMenuItem(item.item_id - 100).pipe(
                  map((menuItem: any) => ({
                      id: item.item_id,
                      name: menuItem.name,
                      price: item.count
                  }))
              );
          });

          // Combine the Observables into a single Observable that emits an array of menu items
          return forkJoin(observables);
      })
  ).subscribe((topMenuItems: unknown) => {
    this.topMenuItems = topMenuItems as any[];
  });
}


  /**
   * Navigates to the employee login page.
   */
  goHome(): void {
    this.router.navigate(['/employee-login']);
  }

  /**
   * Toggles the current visual theme of the component.
   */
  switchStyles(): void {
    this.Bionic = !this.Bionic;
  }
}
